  7 DefaultErrorHandler = ErrorHandler.extend({
  8 	handle: function(err, event) {
  9 		if (typeof err == 'object') {
 10 			if (err.Exception == 'RequestInterrupted') {
 11 				return;
 12 			}
 13 			if (err.Exception != undefined)	{
 14 				alert("["+err.Exception+"Exception] "+err.Message);
 15 			} else {
 16 				alert(err);
 17 			}
 18 			return;
 19 		}
 20 		alert("Error caught: "+err);
 21 	}
 22 });

 24 Request = Class.extend(
 25 /** @lends Request# */		
 26 {
 27 	/**
 28 	 * Create a new request.
 29 	 * @param {String} name the name of the page
 30 	 * @param {Object} type reserved
 31 	 * @param {Object} params the parameters associated with the request
 32 	 * @param {hideParams} a list of parameters that will not be displayed in
 33 	 * the URL bar when the request is executed
 34 	 * @class Represents a request
 35 	 * @augments Class
 36 	 * @constructs
 37 	 */
 38 	init: function(name, type, params, hideParams) {
 39 		if (name != undefined)
 40 			name = name.trim();
 41 		this.name = name;
 42 		this.type = type;
 43 		if (params == undefined) {
 44 			params = Array();
 45 		}
 46 		if (hideParams == undefined) {
 47 			hideParams = Array();
 48 		}
 49 		this.params = params;
 50 		this.hiddenParams = hideParams;
 51 		this.setParams(params);
 52 		this.demanded = true;
 53 	},
 54 	
 55 	/**
 56 	 * Check if the request is demanded by the application itself.
 57 	 * @returns {Boolean} the demanding flag
 58 	 */
 59 	isDemanded: function() {
 60 		return this.demanded;
 61 	},
 62 	
 63 	/**
 64 	 * Change the demanding flag of the current request.
 65 	 * Demanded request will be automatically routed.
 66 	 * @param {Boolean} b the demanding flag
 67 	 */
 68 	demand: function(b) {
 69 		this.demanded = b;
 70 	},
 71 	
 72 	/**
 73 	 * Set the value of a specific parameter
 74 	 * @param {String} key the parameter name
 75 	 * @param {String} value the new value
 76 	 */
 77 	setParam: function(key, value) {
 78 		this.params[key] = value;
 79 	},
 80 	
 81 	/**
 82 	 * Change all parameters to a new map
 83 	 * @param {Object} params the new parameters map
 84 	 */
 85 	setParams: function(params) {
 86 		this.params = params;
 87 	},
 88 
 89 	/**
 90 	 * Get the value of a paramter of current request
 91 	 * @param {String} key the parameter
 92 	 * @param {String} defaultValue the default value, if the parameter is not defined
 93 	 * @returns {String} the value of the parameter
 94 	 */
 95 	getParam: function(key, defaultValue) {
 96 		if (this.params[key] == undefined) {
 97 			return defaultValue;
 98 		}
 99 		return this.params[key];
100 	},
101 
102 	/**
103 	 * Get all parameters.
104 	 * @returns {Object} the parameters map
105 	 */
106 	getParams: function() {
107 		return this.params;
108 	},
109 	
110 	/**
111 	 * Change the hash value of current location.
112 	 * @param {String} strToAdd the location after the hash symbol (#)
113 	 */
114 	addHash: function(strToAdd) {
115 		window.location.hash = strToAdd;
116 	},
117 	
118 	/**
119 	 * Change the name of the page represented by this request.
120 	 * @param {String} name the name of the page
121 	 */
122 	setName: function(name) {
123 		this.name = name;
124 	},
125 
126 	/**
127 	 * Get the name of the page represented by this request
128 	 * @returns {String}
129 	 */
130 	getName: function() {
131 		return this.name;
132 	},
133 	
134 	getType: function() {
135 		return this.type;
136 	},
137 	
138 	toString: function() {
139 		return "Request";
140 	}
141 });

143 Request.setProactive = function(b, url) {
144 	if (url == undefined)
145 		url = window.location.hash;
146 	//console.log('set proactive to '+b+' for url: '+url);
147 	//	//console.warn('called by '+Request.setProactive.caller);
148 	if (Request.proactive == undefined)
149 		Request.proactive = {};
150 	Request.proactive[url] = b;
151 };
152 
153 Request.getProactive = function(url) {
154 	if (url == undefined)
155 		url = window.location.hash;
156 	if (Request.proactive == undefined)
157 		Request.proactive = {};
158 	return Request.proactive[url];
159 };

161 RequestHandler = Class.extend(
162 /** @lends RequestHandler# */		
163 {
164 	/**
165 	 * Initialize fields
166 	 * @class Default request handler
167 	 * @augments Class
168 	 * @constructs
169 	 */
170 	init: function() {
171 		this.currentPage = undefined;
172 		this.requestInterrupted = false;
173 		this.autoRouteDefault = true;
174 		this.systemProperties = SingletonFactory.getInstance(Application).getSystemProperties();
175 		this.errorHandler = new DefaultErrorHandler();
176 	},
177 	
178 	/**
179 	 * Change the error handler
180 	 * @param {ErrorHandler} errorHandler the new error handler
181 	 */
182 	setErrorHandler: function(errorHandler) {
183 		this.errorHandler = errorHandler;
184 	},
185 	
186 	/**
187 	 * Route (if needed) and handle a request
188 	 * @param {Request} request the request to be handled
189 	 */
190 	handleRequest: function(request) {
191 		//console.log('current page is '+this.currentPage);
192 		if (this.currentPage != undefined) {
193 			//console.log("Request Interrupted");
194 			this.requestInterrupted = true;
195 		} else {
196 			this.requestInterrupted = false;
197 		}
198 		this.routeRequest(request);
199 		this._handleRequest(request);
200 	},
201 	
202 	_handleRequest: function(request) {
203 		var name = request.getName();
204 		JOOUtils.generateEvent('RequestBeforeHandled', {to: name});
205 		var page = SingletonFactory.getInstance(Page);
206 		this.currentPage = page;
207 		page.setRequest(request);
208 		//console.log('current page begin: '+request.getName());
209 		try 
210 		{
211 			page.onBegin(name);
212 		} 
213 		catch (err) {
214 			log(err);
215 			/*
216 			* display a message similar to 'this applet failed to load. click here to reload it'
217 			*/
218 			this.errorHandler.handle(err, 'onBegin');
219 		}
220 		
221 		//console.log('current page running: '+request.getName());
222 
223 		try 
224 		{
225 			page.run();
226 		} 
227 		catch (err) {
228 			log(err);
229 			/*
230  			* display a message box notify the error
231  			*/
232 			this.errorHandler.handle(err, 'run');
233 		}
234 		
235 		try 
236 		{
237 			page.onEnd();
238 		} 
239 		catch (err) {
240 			log(err);
241 			this.errorHandler.handle(err, 'onEnd');
242 		}
243 
244 		//console.log('current page finished: '+request.getName());
245 		this.currentPage = undefined;
246 
247 		JOOUtils.generateEvent('HtmlRendered');
248 		//		//console.log(currentPage);
249 		if (this.requestInterrupted == true) {
250 			throw {Exception:"RequestInterrupted"};
251 		}
252 	},
253 	
254 	/**
255 	 * Define setter and getter for the window location hash.
256 	 */
257 	prepareForRequest: function() {
258 		if(!("hash" in window.location)) {
259 			window.location.__defineGetter__("hash", function() {
260 				if(location.href.indexOf("#") == -1)
261 					return "";
262 				return location.href.substring(location.href.indexOf("#"));
263 			});
264 			window.location.__defineSetter__("hash", function(v) {
265 				if(location.href.indexOf("#") == -1)
266 					location.href += v;
267 				location.href = location.substring(0,location.href.indexOf("#")) + v;
268 			});
269 		}
270 	},
271 	
272 	/**
273 	 * Create a request based on the current URL
274 	 */
275 	assembleRequest: function() {
276 		var defaultPage = SingletonFactory.getInstance(Application).getSystemProperties().get('page.default', 'Home');
277 		if(window.location.hash == "") {
278 			var request = new Request(defaultPage, null, null, {'page': ''});
279 			return request;
280 		} else {
281 			//console.log("hey!");
282 			var hash = window.location.hash;
283 			hash = hash.substring(1,hash.length);
284 			if (hash.charAt(0) == '!')	{
285 				hash = hash.substring(1, hash.length);
286 			}
287 			var tmp = hash.split("/");
288 			var params = new Array();
289 			var pagename = "";
290 			var i = 0;
291 
292 			while(i<tmp.length) {
293 				if(tmp[i] != "") {
294 					params[tmp[i]] = tmp[i+1];
295 					if(tmp[i] == "page") {
296 						pagename = params[tmp[i]];
297 					}
298 					i = i + 2;
299 				} else {
300 					i = i + 1;
301 				}
302 			}
303 			
304 			var request = new Request(pagename, null, params);
305 			request.demand(false);
306 			return request;
307 		}
308 	},
309 	
310 	/**
311 	 * Modify the URL (i.e the window location) based on the request.
312 	 * @param {Request} request the request to be routed
313 	 */
314 	routeRequest: function(request) {
315 		//if this request neither proactive nor demanded, then there's no point routing it
316 		if (Request.getProactive() == false && request.isDemanded() == false)
317 			return;
318 		var str = "!";
319 		if ( ( request.getName() == undefined || request.getName() == "" ) && this.autoRouteDefault ) {
320 			var pagename = this.systemProperties.get('page.default');
321 			//console.warn('page is undefined! Using default homepage ['+pagename+']');
322 			if (pagename == undefined) {
323 				//console.error('Default page is undefined! I give up for now!');
324 				throw {Exception: "NotFound", Message: "Both default page and parameter page is undefined"};
325 				return undefined;
326 			}
327 			request.setName(pagename);
328 			request.hiddenParams.page = '';
329 		}
330 		request.getParams()['page'] = request.getName();
331 		for(var key in request.getParams()) {
332 			if (request.hiddenParams.hasOwnProperty(key)) {
333 				continue;
334 			}
335 			if (typeof request.getParams()[key] == 'function' || typeof request.getParams()[key] == 'object')	{
336 				continue;
337 			}
338 			value = request.params[key];
339 			if (value != undefined)
340 				str += key+"/"+value+"/";
341 			else
342 				str += key+"/";
343 		}
344 		str = str.substring(0,str.length-1);
345 		if(!("hash" in window.location)) {
346 			window.location.__defineGetter__("hash", function() {
347 				if(location.href.indexOf("#") == -1)
348 					return "";
349 				return location.href.substring(location.href.indexOf("#"));
350 			});
351 			window.location.__defineSetter__("hash", function(v) {
352 				if(location.href.indexOf("#") == -1)
353 					location.href += v;
354 				location.href = location.substring(0,location.href.indexOf("#")) + v;
355 			});
356 		}
357 		//mark the current request as Proactive, so it won't trigger another history event
358 		Request.setProactive(true, '#!'+str);
359 		window.location.hash = str;
360 	},
361 	
362 	toString: function() {
363 		return "RequestHandler";
364 	}
365 });
