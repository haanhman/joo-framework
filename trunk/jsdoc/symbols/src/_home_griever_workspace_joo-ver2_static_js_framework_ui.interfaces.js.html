1 /**
  2  * @class An interface which allows UI Component to be selectable.
  3  * @interface
  4  */
  5 SelectableInterface = InterfaceImplementor.extend({
  6 	
  7 	implement: function(obj) {
  8 		
  9 		/**
 10 		 * A protected method, define the behavior of the selection.
 11 		 * By default, it does nothing. Subclass can override it to
 12 		 * change the behavior.
 13 		 */
 14 		obj.prototype._select = obj.prototype._select || function() {};
 15 		
 16 		/**
 17 		 * A protected method, define the behavior of the de-selection.
 18 		 * By default, it does nothing. Subclass can override it to
 19 		 * change the behavior.
 20 		 */
 21 		obj.prototype._deselect = obj.prototype._deselect || function() {};
 22 		
 23 		/**
 24 		 * Select the component. Add the component to the stage's list of selection.
 25 		 * @methodOf SelectableInterface#
 26 		 * @name select
 27 		 * @param {Boolean} isMultiSelect whether this is a multi-selection
 28 		 */
 29 		obj.prototype.select = obj.prototype.select || function(isMultiSelect) {
 30 			this.dispatchEvent('beforeSelected');
 31 			if (this.blockSelect != true) {
 32 				this.stage.addSelectedObject(this, isMultiSelect);
 33 				this.dispatchEvent('afterSelected');
 34 			}
 35 		},
 36 		
 37 		/**
 38 		 * Deselect the component. Remove the component from the stage's list of selection.
 39 		 * @methodOf SelectableInterface#
 40 		 * @name deselect
 41 		 * @param {Boolean} isMultiSelect whether this is a multi-selection
 42 		 */
 43 		obj.prototype.deselect = obj.prototype.deselect || function() {
 44 			this.stage.removeSelectedObject(this);
 45 		},
 46 		
 47 		/**
 48 		 * This method is called internally by the Stage, before the 
 49 		 * component is actually selected.
 50 		 * @methodOf SelectableInterface#
 51 		 * @name stageSelect
 52 		 */
 53 		obj.prototype.stageSelect = obj.prototype.stageSelect || function() {
 54 			this.access().addClass('selected');
 55 			this._select();
 56 			this.dispatchEvent('selected');
 57 		},
 58 		
 59 		/**
 60 		 * This method is called internally by the Stage, before the 
 61 		 * component is actually deselected.
 62 		 * @methodOf SelectableInterface#
 63 		 * @name stageDeselect
 64 		 */
 65 		obj.prototype.stageDeselect = obj.prototype.stageDeselect || function() {
 66 			this.access().removeClass('selected');
 67 			this._deselect();
 68 			this.dispatchEvent('deselected');
 69 		};
 70 	}
 71 });

 73 /**
 74  * @class An interface which allows UI Component to be draggable.
 75  * @interface
 76  */
 77 DraggableInterface = InterfaceImplementor.extend({
 78 	
 79 	implement: function(obj){
 80 		
 81 		/**
 82 		 * Make the component draggable. It position will be changed to absolute.
 83 		 * @methodOf DraggableInterface#
 84 		 * @name draggable
 85 		 * @param {Object} param the parameters passed to the draggable engine
 86 		 */
 87 		obj.prototype.draggable = obj.prototype.draggable || function(param) {
 88 			this.access().draggable(param);
 89 			this.setStyle('position', 'absolute');
 90 		};
 91 		
 92 		/**
 93 		 * A method called before the component is dragged. Override this method
 94 		 * to change the behavior.
 95 		 * @methodOf DraggableInterface#
 96 		 * @name beforeStartDragging
 97 		 * @param e the event
 98 		 */
 99 		obj.prototype.beforeStartDragging = obj.prototype.beforeStartDragging || function(e) {};
100 
101 		/**
102 		 * Enable dragging.
103 		 * @methodOf DraggableInterface#
104 		 * @name startDrag
105 		 * @param {Object} param the parameters passed to the draggable engine
106 		 */
107 		obj.prototype.startDrag = obj.prototype.startDrag || function(param) {
108 			var _self = this;
109 			this.setStyle('position', 'absolute');
110 			this.access().draggable({
111 				drag: function(e) {
112 					_self.isDragging = true;
113 				},
114 				beforeStart: function(e) {
115 					_self.beforeStartDragging(e);
116 				}
117 			});
118 			this.access().draggable(param || "enable");
119 		};
120 		
121 		/**
122 		 * Disable dragging.
123 		 * @methodOf DraggableInterface#
124 		 * @name stopDrag
125 		 */
126 		obj.prototype.stopDrag = obj.prototype.stopDrag || function(){
127 			this.access().draggable("disable");
128 		};
129 	},
130 });

167 RotateIcon = Sketch.extend({
168 	
169 	setupBase: function(config) {
170 		this.container = config.container;
171 		this._super(config);
172 	},
173 	
174 	setupDomObject: function(config) {
175 		this._super(config);
176 		this.baseRotate = new Sketch();
177 		this.baseRotate.access().addClass("joo-base-rotate");
178 		this.alignRotate = new Sketch();
179 		this.alignRotate.access().addClass("joo-align-rotate");
180 		
181 		this.initialControlX = 50;
182 		this.initialControlY = 5.5;
183 		this.controlRotate = new Sketch();
184 		this.controlRotate.access().addClass("joo-control-rotate");
185 		this.controlRotate.setLocation(this.initialControlX, this.initialControlY);
186 //		DraggableWrapper.wrap(this.controlRotate);
187 //		this.controlRotate.draggable();
188 		
189 		var baseX = this.baseRotate.getX();
190 		var baseY = this.baseRotate.getY();
191 		var controlX = this.controlRotate.getX();
192 		var controlY = this.controlRotate.getY();
193 		if (controlX == 0) controlX = this.initialControlX;
194 		if (controlY == 0) controlY = this.initialControlY;
195 		var distance = MathUtil.getDistance({x: baseX+9, y: baseY+9}, {x: controlX+4.5, y: controlY+4.5});
196 		this.alignRotate.setWidth(distance);
197 		
198 		this.addChild(this.baseRotate);
199 		this.addChild(this.alignRotate);
200 		this.addChild(this.controlRotate);
201 		this.dragging = false;
202 	},
203 	
204 	registerEvent: function() {
205 		if (!this.dragging) {
206 			this.addEventListener('mousedown', function(e) {
207 				e.stopPropagation();
208 				this.container.changeTransformOrigin('center');
209 				$(window).bind("mousemove", {_self: this}, this.mouseMoveHandler);
210 				this.addEventListener("mousemove", this.mouseMoveHandler);
211 				this.dragging = true;
212 			});
213 		}
214 	},
215 	
216 	unregisterEvent: function() {
217 		if (this.dragging) {
218 			$(window).unbind("mousemove", this.mouseMoveHandler);
219 			this.removeEventListener("mousemove", this.mouseMoveHandler);
220 			this.dragging = false;
221 		}
222 	},
223 	
224 	updateArea: function(e) {
225 		this.newW = e.w || this.newW;
226 		this.newH = e.h || this.newH;
227 		
228 		var ctrlBtnOffset = -9;
229 		this.setLocation(this.newW/2 + ctrlBtnOffset, this.newH/2 + ctrlBtnOffset);
230 	},
231 	
232 	mouseMoveHandler: function(e) {
233 		var _self = this;
234 		if (e.data != undefined) {
235 			_self = e.data._self || this;
236 		}
237 		var baseX = _self.baseRotate.transformedOffset().x;
238 		var baseY = _self.baseRotate.transformedOffset().y;
239 		var controlX = e.pageX;
240 		var controlY = e.pageY;
241 		var angle = MathUtil.getAngle({x: baseX+9, y: baseY+9}, {x: controlX, y: controlY}, true);
242 		_self.dispatchEvent('areaChanged', {a: angle, base: {x: baseX+9, y: baseY+9}});
243 		//_self.controlRotate.setX(MathUtil.getDistance({x: baseX+9, y: baseY+9}, {x: controlX, y: controlY}));
244 	}

247 ResizeIcon = Sketch.extend({
248 	
249 	setupBase: function(config) {
250 		this.pos = config.pos;
251 		this.container = config.container;
252 		this._super(config);
253 	},
254 	
255 	setupDomObject: function(config) {
256 		this._super(config);
257 		this.access().addClass("unselectable");
258 		this.access().addClass("joo-move-icon");
259 		this.setStyle("cursor", this.pos+"-resize");
260 		this.dragging = false;
261 	},
262 	
263 	registerEvent: function() {
264 		this.addEventListener('mousedown', function(e) {
265 			e.stopPropagation();
266 			this.container.changeTransformOrigin('topLeft');
267 			$(window).bind("mousemove", {_self: this}, this.mouseMoveHandler);
268 			this.addEventListener("mousemove", this.mouseMoveHandler);
269 			this.dragging = true;
270 		});	
271 	},
272 	
273 	unregisterEvent: function() {
274 		if (this.dragging) {
275 			$(window).unbind("mousemove", this.mouseMoveHandler);
276 			this.removeEventListener("mousemove", this.mouseMoveHandler);
277 			this.dragging = false;
278 		}
279 	},
280 	
281 	updateArea: function(e) {
282 		this.newW = e.w || this.newW;
283 		this.newH = e.h || this.newH;
284 		
285 		var ctrlBtnOffset = -4;
286 		switch(this.pos) {
287 		case 'n':
288 			this.setLocation(this.newW/2 + ctrlBtnOffset, ctrlBtnOffset);
289 			break;
290 		case 'ne':
291 			this.setLocation(this.newW + ctrlBtnOffset, ctrlBtnOffset);
292 			break;
293 		case 'e':
294 			this.setLocation(this.newW + ctrlBtnOffset, this.newH/2 + ctrlBtnOffset);
295 			break;
296 		case 'se':
297 			this.setLocation(this.newW + ctrlBtnOffset, this.newH + ctrlBtnOffset);
298 			break;
299 		case 's':
300 			this.setLocation(this.newW/2 + ctrlBtnOffset, this.newH + ctrlBtnOffset);
301 			break;
302 		case 'sw':
303 			this.setLocation(ctrlBtnOffset, this.newH + ctrlBtnOffset);
304 			break;
305 		case 'w':
306 			this.setLocation(ctrlBtnOffset, this.newH/2 + ctrlBtnOffset);
307 			break;
308 		case 'nw':
309 			this.setLocation(ctrlBtnOffset, ctrlBtnOffset);
310 			break;
311 		default:
312 		}
313 	},
314 	
315 	mouseMoveHandler: function(e) {
316 		var _self = this;
317 		if (e.data != undefined) {
318 			_self = e.data._self || this;
319 		}
320 		var method = 'mouseMoveHandler' + _self.pos.toUpperCase();
321 		if (typeof _self[method] != 'undefined') {
322 			_self[method].call(_self, e);
323 			_self.dragging = true;
324 		}
325 	},
326 	
327 	mouseMoveHandlerN: function(e) {
328 		var pos = this.getContainerDeltaPosition(e);
329 		var angle = this.container.getRotation() * Math.PI / 180;
330 		var incx = pos.y * Math.sin(angle);
331 		var incy = pos.y * Math.cos(angle);
332 		this.newH = this.container.getHeight() - pos.y;
333 		this.dispatchEvent('areaChanged', {x: this.container.getX() - incx, y: this.container.getY() + incy, h: this.newH});
334 	},
335 	
336 	mouseMoveHandlerNW: function(e) {
337 		var pos = this.getContainerDeltaPosition(e);
338 		var baseOff = this.container._parent.offset();
339 		this.newW = this.container.getWidth() - pos.x;
340 		this.newH = this.container.getHeight() - pos.y;
341 		this.dispatchEvent('areaChanged', {x: e.pageX - baseOff.x, y: e.pageY - baseOff.y, h: this.newH, w: this.newW});
342 	},
343 	
344 	mouseMoveHandlerW: function(e) {
345 		var pos = this.getContainerDeltaPosition(e);
346 		var angle = this.container.getRotation() * Math.PI / 180;
347 		var incx = pos.x * Math.cos(angle);
348 		var incy = pos.x * Math.sin(angle);
349 		this.dispatchEvent('areaChanged', {x: this.container.getX() + incx, y: this.container.getY() + incy, w: this.container.getWidth() - pos.x});
350 	},
351 	
352 	mouseMoveHandlerSW: function(e){
353 		var pos = this.getContainerDeltaPosition(e);
354 		var angle = this.container.getRotation() * Math.PI / 180;
355 		this.newW = -pos.x+this.container.getWidth();
356 		this.newH = pos.y;
357 		var incx = pos.x * Math.cos(angle);
358 		var incy = pos.x * Math.sin(angle);
359 		this.dispatchEvent('areaChanged', {x: this.container.getX()+incx, y: this.container.getY()+incy, w: this.newW, h: this.newH});
360 	},
361 
362 	mouseMoveHandlerS: function(e) {
363 		var pos = this.getContainerDeltaPosition(e);
364 		this.newH = pos.y;
365 		this.dispatchEvent('areaChanged', {h: this.newH});
366 	},
367 	
368 	mouseMoveHandlerSE: function(e) {
369 		var pos = this.getContainerDeltaPosition(e);
370 		this.newW = pos.x;
371 		this.newH = pos.y;
372 		this.dispatchEvent('areaChanged', {w: this.newW, h: this.newH});
373 	},
374 	
375 	mouseMoveHandlerE: function(e){
376 		var pos = this.getContainerDeltaPosition(e);
377 		this.newW = pos.x;
378 		this.dispatchEvent('areaChanged', {w: this.newW});
379 	},
380 	
381 	mouseMoveHandlerNE: function(e){
382 		var pos = this.getContainerDeltaPosition(e);
383 		var angle = this.container.getRotation() * Math.PI / 180;
384 		this.newW = pos.x;
385 		this.newH = -pos.y+this.container.getHeight();
386 		var incx = pos.y * Math.sin(angle);
387 		var incy = pos.y * Math.cos(angle);
388 		
389 		this.dispatchEvent('areaChanged', {h: this.newH, w: this.newW, x: this.container.getX() - incx, y: this.container.getY() + incy});
390 	},
391 	
392 	getContainerDeltaPosition: function(e) {
393 		var selfPos = this.container.offset();
394 		return getPositionInRotatedcoordinate({
395 			x : e.pageX - selfPos.x,
396 			y : e.pageY - selfPos.y
397 		}, Math.PI * this.container.getRotation() / 180);
398 	},
399 	
400 	getContainerOffsetPosition: function(newW, newH) {
401 		var newCenterPoint = getPositionFromRotatedCoordinate({
402 			x : (newW - this.container.getWidth()) / 2,
403 			y : (newH - this.container.getHeight()) / 2
404 		}, Math.PI * this.container.getRotation() / 180, this.container.getRotationCenterPoint());
405 		var res = getPositionFromRotatedCoordinate({
406 			x : - newW / 2,
407 			y : - newH / 2
408 		}, Math.PI * this.container.getRotation() / 180, newCenterPoint);
409 		return res;
410 	}
411 });

464 ResizableInterface = InterfaceImplementor.extend({
465 	
466 	implement: function(obj) {
467 
468 		obj.prototype.updateArea = obj.prototype.updateArea || function(e) {
469 			this.canvasW = this.canvasW || e.w;
470 			this.canvasH = this.canvasH || e.h;
471 			
472 			this.doUpdateArea(e);
473 			e.w = this.access().outerWidth();
474 			e.h = this.access().outerHeight();
475 			this.resizeset.updateArea(e);
476 		};
477 		
478 		obj.prototype.doUpdateArea = obj.prototype.doUpdateArea || function(e) {
479 			if (e.a != undefined) {
480 				this.setRotation(e.a);
481 			} else {
482 				var deltaW = this.access().outerWidth() - this.getWidth();
483 				var deltaH = this.access().outerHeight() - this.getHeight();
484 				if (e.x) {
485 					this.setX(e.x);
486 				}
487 				if (e.y)
488 					this.setY(e.y);
489 				if (e.h)
490 					this.setHeight(e.h - deltaH);
491 				if (e.w)
492 					this.setWidth(e.w - deltaW);
493 			}
494 		};
495 		
496 		obj.prototype.showResizeControl = obj.prototype.showResizeControl || function(){
497 			this.resizeset.show();
498 		},
499 		
500 		obj.prototype.hideResizeControl = obj.prototype.hideResizeControl || function(){
501 			this.resizeset.hide();
502 		},
503 		
504 		obj.prototype.onDeleteHandler = obj.prototype.onDeleteHandler || function(e) {
505 			this.dispose();
506 		};
507 		
508 		obj.prototype.onMouseUpHandler = obj.prototype.onMouseUpHandler || function(e) {};
509 		
510 		obj.prototype.onMouseDownHandler = obj.prototype.onMouseDownHandler || function(e) {};
511 		
512 		obj.prototype.beginEditable = obj.prototype.beginEditable || function(defaultW, defaultH, resizeIcon, includeRotate) {
513 			this.rotationCenter = {
514 				x: 0.5,
515 				y: 0.5
516 			};
517 			if (resizeIcon == undefined)
518 				resizeIcon = ['n', 'nw', 'w', 'sw', 's', 'e', 'se', 'ne'];
519 			this.defaultW = defaultW || 150;
520 			this.defaultH = defaultH || 150;
521 			
522 			this.resizeset = new ResizeIconSet(this, resizeIcon, {minW: 0, minH: 0, maxW: Number.MAX_VALUE, maxH: Number.MAX_VALUE});
523 			if (includeRotate)
524 				this.resizeset.addChild(new RotateIcon());
525 			this.setStyle('position', 'absolute');
526 			
527 			this.access().addClass('joo-init-transform');
528 			this.access().addClass("joo-editable-component");
529 			
530 			var _self = this;
531 			
532 			$(document).bind("mouseup",function(){
533 				_self.changeTransformOrigin('center');
534 				_self.onMouseUpHandler();
535 				_self.resizeset.unregisterEvent();
536 			});
537 
538 			for(var i=0; i<this.resizeset.getButtons().length;i++) {
539 				this.resizeset.getButtons()[i].addEventListener('mousedown', function(e) {
540 					_self.stopDrag();
541 					_self.onMouseDownHandler(e);
542 					_self.dispatchEvent('resizestart');
543 				});
544 				
545 				this.resizeset.getButtons()[i].addEventListener('mouseup', function(e) {
546 					_self.dispatchEvent('resizestop');
547 				});
548 				
549 				this.resizeset.getButtons()[i].addEventListener('stylechange', function(e) {
550 					e.stopPropagation();
551 				});
552 				
553 				this.resizeset.getButtons()[i].addEventListener('areaChanged', function(e) {
554 					if (e.w != undefined) {
555 						if (e.w > _self.maxW)
556 							e.w = _self.maxW;
557 						else if (e.w < _self.minW)
558 							e.w = _self.minW;
559 					}
560 					
561 					if (e.h!= undefined) {
562 						if (e.h > _self.maxH)
563 							e.h = _self.maxH;
564 						else if (e.h < _self.minH)
565 							e.h = _self.minH;
566 					}
567 					
568 					if (e.x != undefined) {
569 						var minX = _self.access().offset().x + _self.getWidth() - _self.maxW;
570 						var maxX = _self.access().offset().x + _self.getWidth() - _self.minW;
571 						if (e.x < minX)
572 							e.x = minX;
573 						else if (e.x > maxX)
574 							e.x = maxX;
575 					}
576 					
577 					if (e.y != undefined) {
578 						var minY = _self.offset().y + _self.getHeight() - _self.maxH;
579 						var maxY = _self.offset().y + _self.getHeight() - _self.minH;
580 						if (e.y < minY)
581 							e.y = minY;
582 						else if (e.y > maxY)
583 							e.y = maxY;
584 					}
585 					_self.updateArea(e);
586 				});
587 			}
588 			this.resizeset.registerEvent();
589 
590 			for(var i=0; i<this.resizeset.getButtons().length; i++) {
591 				this.addChild(this.resizeset.getButtons()[i]);
592 			}
593 			this.updateArea({w: this.defaultW, h: this.defaultH});
594 		};
595 	}
596 });

699 Preset = Class.extend({
700 
701 	init: function() {
702 		this.name = "Preset";
703 		this.changed = {};
704 		this.oldValue = {};
705 	},
706 	
707 	apply: function(obj) {
708 		for(var i in this.changed) {
709 			this.oldValue[i] = obj.getStyle(i);
710 			obj.setStyle(i, this.changed[i]);
711 			this.changed[i] = obj.getStyle(i);
712 		}
713 	},
714 	
715 	revert: function(obj) {
716 		for(var i in this.changed) {
717 			var style = obj.getStyle(i);
718 			if (style == this.changed[i]) {
719 				obj.setStyle(i, this.oldValue[i]);
720 			}
721 		}
722 	},
723 	
724 	getName: function() {
725 		return this.name;
726 	},
727 	
728 	toString: function() {
729 		return this.name;
730 	}
731 });
