1 //micro template based on John Resg code
  2 function tmpl(tmpl_id,data){
  3 	try {
  4 		if ( typeof tmpl.cache == 'undefined' ) {
  5 			tmpl.cache = new Array();
  6 	    }
  7 		if( tmpl.cache[tmpl_id]!=null ){
  8 			var fn = tmpl.cache[tmpl_id];
  9 			return fn(data);		
 10 		}
 11 		str = document.getElementById(tmpl_id).innerHTML;
 12 		str = str.replace(/\\/g, "@SPC@");
 13 		str = str.replace(/'/g, "'");
 14 		fnStr = "var p=[],print=function(){p.push.apply(p,arguments);};" +
 15 	    
 16 	    // Introduce the data as local variables using with(){}
 17 	    "with(obj){p.push('" +
 18 	    
 19 	    // Convert the template into pure JavaScript
 20 	    str
 21 	      .replace(/[\r\t\n]/g, " ")
 22 	      .split("<%").join("\t")
 23 	      .replace(/((^|%>)[^\t]*)'/g, "$1\r")
 24 	      .replace(/\t=(.*?)%>/g, "',$1,'")
 25 	      .split("\t").join("');")
 26 	      .split("%>").join("p.push('")
 27 	      .split("\r").join("\\'")
 28 	  + "');}return p.join('');";
 29 		fnStr = fnStr.replace(/@SPC@/g, "\\");
 30 		var fn = new Function("obj", fnStr);
 31 		tmpl.cache[tmpl_id] = fn;
 32 		return fn(data);
 33 	} catch (e) {
 34 		log(e+":"+tmpl_id, 'rendering');
 35 		return "";
 36 	}
 37 }
