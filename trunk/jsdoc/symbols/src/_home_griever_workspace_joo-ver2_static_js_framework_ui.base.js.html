 92 EventDispatcher = Class.extend(
 93 /**
 94  * @lends EventDispatcher#
 95  */	
 96 {
 97 
 98 	/**
 99 	 * Create a new EventDispatcher.
100 	 * @class Base class for all event dispatchers (such as DisplayObject)
101 	 * @constructs
102 	 * @augments Class
103 	 */
104 	init: function() {
105 		this.listeners = {};
106 	},
107 	
108 	/**
109 	 * Add a new listener for a specific event.
110 	 * @param {String} event the event to be handled. 
111 	 * @param {Function} handler the event handler. If you want to remove it
112 	 * at a later time, it must not be an anonymous function
113 	 */
114 	addEventListener: function(event, handler) {
115 		if (this.listeners[event] == undefined) {
116 			this.listeners[event] = Array();
117 		}
118 		this.listeners[event].push(handler);
119 	},
120 	
121 	/**
122 	 * Dispatch a event.
123 	 * @param {String} event the event to be dispatched.
124 	 */
125 	dispatchEvent: function(event) {
126 		if (this.listeners != undefined && this.listeners[event] != undefined) {
127 			var handlers = this.listeners[event];
128 			var args = Array();
129 			for(var i=1; i<arguments.length; i++) {
130 				args.push(arguments[i]);
131 			}
132 			for(var i=0;i<handlers.length;i++) {
133 				var result = handlers[i].apply(this, args);
134 				if (result === false)
135 					return;
136 			}
137 		}
138 	},
139 	
140 	/**
141 	 * Remove a handler for a specific event.
142 	 * @param {String} event the event of handler to be removed 
143 	 * @param {Function} handler the handler to be removed
144 	 */
145 	removeEventListener: function(event, handler) {
146 		if (this.listeners[event] != undefined) {
147 			var index = this.listeners[event].indexOf(handler);
148 			if (index != -1)
149 				this.listeners[event].splice(index, 1);
150 		}
151 	},
152 	
153 	toString: function() {
154 		return "EventDispatcher";
155 	},
156 	
157 	setupBase: function(config) {
158 		
159 	}
160 });

719 	changeTransformOrigin: function(option) {
720 		if (this.transformOrigin == option) return;
721 		var pos = this.transformedOffset();
722 		var selfPos = this.virtualNontransformedOffset();
723 		var deltaX = pos.x - selfPos.x;
724 		var deltaY = pos.y - selfPos.y;
725 		
726 		switch (option) {
727 			case 'topLeft':{
728 				this.setStyle('-webkit-transform-origin', '0 0');
729 				this.setLocation(this.getX() + deltaX, this.getY() + deltaY);
730 				break;
731 			}
732 			case 'center': {
733 				this.setStyle('-webkit-transform-origin', "50% 50%");
734 				this.setLocation(this.getX() - deltaX, this.getY() - deltaY);
735 				break;
736 			}
737 			default: return;
738 		}
739 		this.transformOrigin = option;
