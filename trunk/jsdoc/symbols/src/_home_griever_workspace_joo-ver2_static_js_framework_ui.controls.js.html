1 /**
  2  * @class An editable textbox. This component allows user to change the text
  3  * by doubleclicking it, and when it losts user's focus, it also lost
  4  * the editing capabilities.
  5  * <p>It supports additional configuration parameters</p>
  6  * <ul>
  7  * 	<li><code>readonly</code> Whether the component is readonly</li>
  8  * 	<li><code>blurEvent</code> The event when the component losts editing capabilities</li>
  9  * </ul>
 10  * @augments UIComponent
 11  */
 12 JOOText = UIComponent.extend(
 13 /** @lends JOOText# */
 14 {	
 15 	setupDomObject: function(config) {
 16 		this._super(config);
 17 		this.text = new Sketch();
 18 		if (config.lbl)
 19 			this.text.access().html(config.lbl);
 20 
 21 		if (!config.readonly) {
 22 			this.addEventListener('dblclick', function() {
 23 				this.enableEdit(true);
 24 			});
 25 			this.text.addEventListener('dblclick', function() {
 26 				this._parent.enableEdit(true);
 27 			});
 28 			this.text.addEventListener('keyup', function() {
 29 				var old = this._parent.config.lbl;
 30 				this._parent.config.lbl = this.access().html();
 31 				if (old != this._parent.config.lbl)
 32 					this._parent.dispatchEvent('change');
 33 			});
 34 		}
 35 		
 36 		this.text.access().addClass("text");
 37 		if (!config.readonly && config.blurEvent) {
 38 			this.addEventListener(config.blurEvent, function() {
 39 				this.enableEdit(false);
 40 			});
 41 		}
 42 		this.addChild(this.text);
 43 //		this.attachContextMenu();
 44 //		var _self = this;
 45 //		this.getContextMenu().addItem(new JOOMenuItem({label: 'Edit text', command: function() {

 49 	},

 52 	 * Get the value of the text

373 /**
374  * @class A counterpart of <code>HTML SELECT</code> element.
375  * <p>It supports additional configuration parameters:</p>
376  * <ul>
377  * 	<li><code>options</code> Initial options of this select box. It must be an <code>Array</code>
378  * 		which each element is an object with <code>label</code> and <code>value</code> properties.
379  * 	</li>
380  * 	<li><code>selectedIndex</code> The initially selected index, defaults is 0</li>
381  * 	<li><code>selectedValue</code> The initially selected value. Should not present if <code>selectedIndex</code> is already specified.</li>
382  * </ul>
383  * @augments JOOInput
384  */
385 JOOInputSelect = JOOInput.extend({
386 	
387 	setupDomObject: function(config) {
388 		this._super(config);
389 		
390 		this.options = Array();
391 		var options = config.options || {};
392 		for(var i=0; i<options.length; i++) {
393 			this.addOption(options[i]);
394 		}
395 		this.selectedIndex = config.selectedIndex || 0;
396 		if(config.selectedIndex == undefined && config.selectedValue != undefined){
397 			this.selectedIndex = 0;
398 			for(var i=0;i<this.options.length;i++){
399 				if(this.options[i].value == config.selectedValue){
400 					this.selectedIndex = i;
401 					break;
402 				}
403 			}
404 		}
405 		
406 		this.addEventListener("change", function(e) {
407 			if (e != undefined)
408 				this.selectedIndex = e.currentTarget.options.selectedIndex; 
409 		});
410 		this.access().val(config.selectedValue);
411 	},
412 	
413 	/**
414 	 * Add an option to the select box.
415 	 * @param {Object} param new option, with <code>label</code> and <code>value</code> properties.
416 	 */
417 	addOption: function(param){
418 		this.options.push(param);
419 		if (param.order != undefined){
420 			for(var i=this.options.length-2;i>=param.order;i--){
421 				this.options[i] = this.options[i+1];
422 			}
423 		}
424 		this.access().append("<option value='"+param.value+"'>"+param.label+"</option>");
425 	},
426 	
427 	/**
428 	 * Change the value of the select box
429 	 * @param {String} val new value of the select box.
430 	 */
431 	setValue: function(val) {
432 		this.access().val(val);
433 		this.selectedIndex = this.access().find("option:selected").index()-1;
434 		this.dispatchEvent("change");
435 	},
436 
437 	/**
438 	 * Change the value of the select box to an option by its index.
439 	 * @param {Number} idx the index of the option.
440 	 */
441 	setValueByIndex: function(idx) {
442 		this.selectedIndex = idx;
443 		this.access().find("option").eq(idx).attr("selected", "selected");
444 		this.dispatchEvent("change");
445 	},
446 
447 	/**
448 	 * Get the value of the select box.
449 	 * @returns {String} the select box's value.
450 	 */
451 	getValue: function() {
452 		return this.access().val();
453 	},
454 	
455 	/**
456 	 * Refresh the select box.
457 	 */
458 	refresh: function(){
459 		this.access().html(this.toHtml());
460 	},
461 	
462 	toHtml: function(){
463 		return "<select></select>";
464 	}
465 });

635 /**
636  * @class A sprite is a keyframe-based animation object which has a timeline. This is
637  * base class of all animation classes.
638  * <p>It supports additional configuration parameters:</p>
639  * <ul>
640  * 	<li><code>src</code> the background image source, should be a sprite image</li>
641  * 	<li><code>framerate</code> the framerate of the sprite</li>
642  * 	<li><code>loop</code> whether the animation should loop</li>
643  * 	<li><code>horizontalFramesNo</code> the number of frames in horizontal dimension</li>
644  * 	<li><code>verticalFramesNo</code> the number of frames in vertical dimension</li>
645  * 	<li><code>spriteWidth</code> the width of the viewport of sprite</li>
646  * 	<li><code>spriteHeight</code> the height of the viewport of sprite</li>
647  * 	<li><code>speed</code> the relative speed of sprite, e.g 1.5, 2, etc</li>
648  * </ul>
649  * @augments UIComponent
650  */
651 JOOSprite = UIComponent.extend(
652 /** @lends JOOSprite# */
653 {
654 	setupDomObject: function(config) {
655 		this._super(config);
656 		this.src = config.src;
657 		this.framerate = config.framerate || 30;
658 		this.loop = config.loop || false;
659 		this.currentFrame = 0;
660 		this.horizontalFramesNo = config.horizontalFramesNo;
661 		this.verticalFramesNo = config.verticalFramesNo;
662 		this.spriteWidth = config.spriteWidth;
663 		this.spriteHeight = config.spriteHeight;
664 		this.speed = config.speed || 1;
665 		this.stopped = false;
666 	},
667 
668 	/**
669 	 * Play the sprite from <code>start</code> frame to <code>end</code> frame.
670 	 * @param {Number} start the start frame
671 	 * @param {Number} end the end frame
672 	 */
673 	play: function(start, end) {
674 		this.stopped = false;
675 		this.dispatchEvent("frameStart");
676 		this.startFrame = start || 0;
677 		this.endFrame = end;
678 		if(end == undefined){
679 			this.endFrame = this.verticalFramesNo * this.horizontalFramesNo; 
680 		} 
681 		this.currentFrame = this.startFrame;
682 		
683 		this.setWidth(this.spriteWidth);
684 		this.setHeight(this.spriteHeight);
685 		if (this.src)
686 			this.access().css('background-image', 'url('+this.src+')');
687 
688 		this.playFrame();
689 		this._playWithFramerate(this.framerate);
690 	},
691 	
692 	_playWithFramerate: function(framerate) {
693 		framerate *= this.speed;
694 		if (!this.stopped) {
695 			var _self = this;
696 			this.interval = setInterval(function() {
697 				_self.playFrame();
698 			}, parseFloat(1000/framerate));
699 		}
700 	},
701 
702 	/**
703 	 * Change the relative speed of the sprite.
704 	 * @param speed the relative speed of the sprite
705 	 */
706 	setSpeed: function(speed) {
707 		var tempFramerate = this.framerate * speed;
708 		clearInterval(this.interval);
709 		this._playWithFramerate(tempFramerate);
710 	},
711 	
712 	/**
713 	 * Pause the sprite.
714 	 */
715 	pause: function() {
716 		this.dispatchEvent("framePause");
717 		clearInterval(this.interval);
718 	},
719 	
720 	/**
721 	 * Resume the sprite.
722 	 */
723 	resume: function() {
724 		this.dispatchEvent("frameResume");
725 		this._playWithFramerate(this.interval);
726 	},
727 	
728 	/**
729 	 * Stop the sprite.
730 	 */
731 	stop: function() {
732 		this.dispatchEvent("frameStop");
733 		this.stopped = true;
734 	},
735 	
736 	playFrame: function() {
737 		var ended = false;
738 		if (this.currentFrame > this.endFrame) {
739 			if (this.loop) {
740 				this.currentFrame = this.startFrame;
741 			} else {
742 				ended = true;
743 			}
744 		}
745 		if (ended || this.stopped) {
746 			clearInterval(this.interval);
747 			this.stopped = true;
748 			this.dispatchEvent("frameEnded");
749 			return;
750 		}
751 		this.dispatchEvent("frameEnter");
752 		this.onFrame(this.currentFrame);
753 		this.dispatchEvent("frameExit");
754 		this.currentFrame++;
755 	},
756 	
757 	/**
758 	 * This method defines how animation works. Subclass can override it to
