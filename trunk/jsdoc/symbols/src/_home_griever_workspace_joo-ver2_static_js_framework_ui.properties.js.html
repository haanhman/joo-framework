 75 JOOMediaValue = JOOInput.extend({
 76 	
 77 	setupDomObject: function(config) {
 78 		this._super(config);
 79 		this.link = new JOOLink();
 80 		this.mediaBrowser = new JOOMediaBrowser(config);
 81 		this.addChild(this.link);
 82 		this.addChild(this.mediaBrowser);
 83 		this.mediaBrowser.access().hide();
 84 		this.setValue(config.value);
 85 		
 86 		var _self = this;
 87 		this.link.addEventListener('click', function() {
 88 			_self.mediaBrowser.fetch();
 89 			_self.mediaBrowser.access().show();
 90 		});
 91 		this.link.addEventListener('mouseover', function() {
 92 			showtip(_self.getValue());
 93 		});
 94 		this.link.addEventListener('mouseout', function() {
 95 			hidetip();
 96 		});
 97 		this.mediaBrowser.close = function() {
 98 			this.access().hide();
 99 		};
100 		this.mediaBrowser.addEventListener('change', function() {
101 			_self.onchange(this.getValue());
102 		});
103 	},
104 	
105 	onchange: function(value) {
106 		this.mediaBrowser.close();
107 		this.setValue(value);
108 	},
109 
110 	setValue: function(value) {
111 		this._super(value);
112 		if (value == undefined)
113 			value = "Unspecified";
114 		this.link.access().html(value);
115 		this.dispatchEvent('change');
116 	},
117 	
118 	toHtml: function() {
119 		return "<div></div>";
120 	}
121 });

188 JOOPropertiesDialog = JOODialog.extend({
189 	
190 	setupDomObject: function(config) {
191 		this._super(config);
192 		this.setTitle('Properties');
193 		this.getContentPane().setLayout('vertical');
194 		this.supportedProperties = this.supportedProperties || [];
195 		this.propertyMappings = this.propertyMappings || {
196 			'colorpicker': JOOColorProperty,
197 			'media': JOOMediaProperty,
198 			'slider': JOOSliderProperty,
199 			'select': JOOSelectProperty,
200 			'text': JOOTextProperty,
201 			'textarea': JOOTextAreaProperty
202 		};
203 		
204 		this.generatePropertyElements();
205 		this.reloadList();
206 	},
207 	
208 	removeTarget: function(target) {
209 		if (target != undefined) {
210 			target.removeEventListener(this.onTargetStyle);
211 		}
212 		if (this.target == target) {
213 			this.target = undefined;
214 			this.reloadList();
215 		}
216 	},
217 	
218 	setTarget: function(target) {
219 		if (this.target != undefined) {
220 			this.target.removeEventListener(this.onTargetStyle); 
221 		}
222 		if (this.target != target) {
223 			if(this.target){
224 				currentTargetId = this.target.getId();
225 			}
226 			this.target = target;
227 			this.reloadList();
228 		}
229 	},
230 	
231 	onTargetStyle: function(e) {
232 		//generateEvent('ObjectStyleChanged', e);
233 	},
234 	
235 	updateList: function(e) {
236 		var supported = undefined;
237 		if (this.target)
238 			supported = this.target.getSupportedProperties();
239 		for(var i in this.props) {
240 			if (e.indexOf(this.props[i].options.property.name) == -1 || !this.target || !supported || (supported.indexOf(i) == -1 && supported != "all")) {
241 				
242 			} else {
243 				var orig = this.props[i].element.onchange;
244 				this.props[i].element.onchange = function() {};
245 				this.props[i].element.setValue(this.target.getProperty(this.props[i].options.property));
246 				this.props[i].element.onchange = orig;
247 			}
248 		}
249 	},
250 	
251 	reloadList: function() {
252 		var supported = undefined;
253 		
254 		if (this.target) {
255 			this.target._parent.addEventListener('stylechange', this.onTargetStyle);
256 			this.target.addEventListener('stylechange', this.onTargetStyle);
257 			this.target._parent.addEventListener('dragstop', this.onTargetStyle);
258 			supported = this.target.getSupportedProperties();
259 		}
260 		for(var i in this.props) {
261 			if (!this.target || !supported || (supported.indexOf(i) == -1 && supported != "all")) {
262 				if (this.props[i].options.autohide) {
263 					this.props[i].element.access().hide();
264 					this.props[i].element.setAttribute('hide', '');
265 				}
266 				this.props[i].element.disable(true);
267 			} else {
268 				var _self = this;
269 				this.props[i].element.access().removeAttr('hide');
270 				this.props[i].element.access().show();
271 				
272 				this.props[i].element.disable(false);
273 				this.props[i].element.targetProperty = this.props[i].options.property;
274 				this.props[i].element.setValue(this.target.getProperty(this.props[i].options.property));
275 				this.props[i].element.onchange = function() {
276 					_self.target.setProperty(this.targetProperty, this.getValue());
277 					_self.dispatchEvent('change');
278 				};
279 			}
280 		}
281 		
282 		for(var i=0; i<this.accords.length; i++) {
283 			var len = this.accords[i].getContentPane().access().children(':not([hide])').length;
284 			if (len > 0)
285 				this.accords[i].access().show();
286 			else
287 				this.accords[i].access().hide();
288 		}
289 	},
290 	
291 	generatePropertyElements: function() {
292 		this.props = Array();
293 		this.accords = Array();
294 		
295 		var props = this.supportedProperties;
296 		this.tab = new JOOTabbedPane({width: '98%'});
297 		this.tab.access().addClass('properties-tab');
298 		for(var i=0; i<props.length; i++) {
299 			var sketch = new Sketch();
300 			var cats = props[i].categories;
301 			for(var j=0; j<cats.length; j++) {
302 				var accord = new JOOAccordion({lbl: cats[j].category});
303 				if (cats[j].options != undefined) {
304 					for(var k=0; k<cats[j].options.length; k++) {
305 						var opt = cats[j].options[k];
306 						opt.config.lbl = opt.name;
307 						var element = new this.propertyMappings[opt.type](opt.config);
308 						element.onchange = function() {};
309 						element.addEventListener('change', function() {
310 							this.onchange();
311 						});
312 						accord.getContentPane().addChild(element);
313 						this.props[opt.id] = {element: element, options: opt};
314 					}
315 				}
316 				this.accords.push(accord);
317 				sketch.addChild(accord);
318 			}
319 			this.tab.addTab(props[i].section, sketch);
320 			this.tab.setTab(0);
321 		}
322 		this.getContentPane().addChild(this.tab);
323 	}	
324 }).implement(DraggableInterface);
