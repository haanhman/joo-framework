1 JOOService = EventDispatcher.extend({
  2 	
  3 	init: function(endpoint, method) {
  4 		this._super();
  5 		this.name = "DefaultService";
  6 		this.endpoint = endpoint || "";
  7 		this.method = method || "get";
  8 	},
  9 	
 10 	run: function(params) {
 11 		var _self = this;
 12 		this.onAjax(this.endpoint, params, this.method, {
 13 			onSuccess: function(ret) {
 14 				ret = _self.parse(ret);
 15 				_self.dispatchEvent('success', ret);
 16 				JOOUtils.generateEvent('ServiceSuccess', this.name, ret);
 17 			},
 18 			onFailure: function(msg) {
 19 				msg = _self.parseError(msg);
 20 				_self.dispatchEvent('failure', msg);
 21 				JOOUtils.generateEvent('ServiceFailure', this.name, msg);
 22 			}
 23 		});
 24 	},
 25 	
 26 	parse: function(ret) {
 27 		return ret;
 28 	},
 29 	
 30 	parseError: function(msg) {
 31 		return msg;
 32 	},
 33 	
 34 	getEndPoint: function() {
 35 		return this.endpoint;
 36 	}
 37 }).implement(AjaxInterface);

 39 /**
 40  * @class A model which supports property change event.
 41  * @augments EventDispatcher
 42  */
 43 JOOModel = EventDispatcher.extend({
 44 	
 45 	bindings: function(obj) {
 46 		obj = obj || this;
 47 		for(var i in obj) {
 48 			this._bindings(obj, i);
 49 		}
 50 	},
 51 	
 52 	_bindings: function(obj, i) {
 53 		if (i == 'className' || i == 'ancestors' || i == 'listeners') 
 54 			return;
 55 		if (typeof obj[i] != 'function') {
 56 			if (obj[i] instanceof Object || obj[i] instanceof Array ) {
 57 				if (obj[i] instanceof Array) {
 58 					this.bindForArray(obj[i]);
 59 				}
 60 				this.bindings(obj[i]); //recursively bind
 61 			}
 62 			this.bindForValue(obj, i);
 63 		}
 64 	},
 65 	
 66 	bindForArray: function(obj) {
 67 		var _self = this;
 68 	    var length = obj.length;
 69 	    obj.__defineGetter__("length", function() {
 70 			return length;
 71 		});
 72 	    this.hookUp(obj, 'push', function(item) {
 73 	    	_self._bindings(obj, obj.length-1);
 74 	    });
 75 	    this.hookUp(obj, 'pop');
 76 	    this.hookUp(obj, 'splice', function() {
 77 	    	for(var i=2; i<arguments.length; i++) {
 78 	    		_self._bindings(obj, obj.length-arguments.length-i);
 79 	    	}
 80 	    });
 81 	},
 82 	
 83 	hookUp: function(obj, fn, callback) {
 84 		var _self = this;
 85 		var orig = obj[fn];
 86 	    obj[fn] = function() {
 87 	    	orig.apply(obj, arguments);
 88 	    	callback.apply(undefined, arguments);
 89 	    	_self.dispatchEvent('change');
 90 	    };
 91 	},
 92 	
 93 	bindForValue: function(obj, i) {
 94 		var _self = this;
 95 		var prop = "_"+i;
 96 		obj[prop] = obj[i];
 97 		obj[i] = undefined;
 98 		delete obj[i];
 99 		
100 		if (!obj.__lookupGetter__(i)) {
101 			obj.__defineGetter__(i, function() {
102 		        return obj[prop];
103 		    });
104 		}
105 		if (!obj.__lookupSetter__(i)) {
106 			obj.__defineSetter__(i, function(val) {
107 		        obj[prop] = val;
108 		        _self.dispatchEvent('change');
109 		    });
110 		}
111 	}
112 });

114 /**
115  * Create or extend model from ordinary object
116  * @param {Object} obj the object 
117  * @param {JOOModel} model existing model 
118  * @returns the result model
119  */
120 JOOModel.from = function(obj, model) {
121 	model = model || new JOOModel();
122 	for(var i in obj) {
123 		model[i] = obj[i];
124 	}
125 	model.bindings();
126 	return model;
127 };
