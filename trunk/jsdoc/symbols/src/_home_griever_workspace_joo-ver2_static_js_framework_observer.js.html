1 /**
  2  * @class Used for formalizing the observer design pattern,
  3  * especially in an event-based application
  4  * @interface
  5  */
  6 ObserverInterface = InterfaceImplementor.extend({
  7 	
  8 	implement: function(obj)	{
  9 		/**
 10 		 * Called when the observer is notified of an event by the {@link Subject}.
 11 		 * The default implementation forward the request
 12 		 * @methodOf ObserverInterface#
 13 		 * @name notify 
 14 		 * @param {String} eventName the event name
 15 		 * @param {Object} eventData the event data
 16 		 * @returns {Boolean} whether the event is interested by this observer or not.
 17 		 */
 18 		obj.prototype.notify = obj.prototype.notify || function(eventName, eventData)	{
 19 			var methodName = "on"+eventName;
 20 			if (typeof this[methodName] != 'undefined')	{
 21 				var method = this[methodName];
 22 				method.call(this, eventData);
 23 				return true;
 24 			}
 25 			return false;
 26 		};
 27 		
 28 		/**
 29 		 * Register this observer with the {@link Subject}.
 30 		 * @methodOf ObserverInterface#
 31 		 * @name registerObserver
 32 		 */
 33 		obj.prototype.registerObserver = obj.prototype.registerObserver || function()	{
 34 			var subject = SingletonFactory.getInstance(Subject);
 35 			subject.attachObserver(this);
 36 		};
 37 		
 38 		/**
 39 		 * Unregister this observer with the {@link Subject}.
 40 		 * @methodOf ObserverInterface#
 41 		 * @name unregisterObserver
 42 		 */
 43 		obj.prototype.unregisterObserver = obj.prototype.unregisterObserver || function()	{
 44 			var subject = SingletonFactory.getInstance(Subject);
 45 			subject.detachObserver(this);
 46 		};
 47 	}
 48 });
 49 
 50 Subject = Class.extend(
 51 /** @lends Subject# */	
 52 {
 53 	
 54 	/**
 55 	 * Initialize observers
 56 	 * @class <code>Subject</code> is the central of Observer pattern. It maintains a list
 57 	 * of observers, and notifies them automatically of new events. <code>Subject</code> is
 58 	 * a <code>singleton</code> class.
 59 	 * @augments Class
 60 	 * @constructs
 61 	 */
 62 	init: function()	{
 63 		this.observers = Array();
 64 	},
 65 	
 66 	/**
 67 	 * Attach an observer
 68 	 * @param {ObserverInterface} observer the observer to be attached
 69 	 */
 70 	attachObserver: function(observer)	{
 71 		this.observers.push(observer);
 72 	},
 73 	
 74 	/**
 75 	 * Detach an observer
 76 	 * @param {ObserverInterface} observer the observer to be detached
 77 	 */
 78 	detachObserver: function(observer)	{
 79 		if (observer == undefined)
 80 			return;
 81 		var index = this.observers.indexOf(observer);
 82 		if (index > 0)	{
 83 			this.observers.splice(index, 1);
 84 		}
 85 	},
 86 	
 87 	/**
 88 	 * Notify an event to all observers
 89 	 * @param {String} eventName the name of the event which should contains characters only
 90 	 * @param {Object} eventData the data associated with the event
 91 	 */
 92 	notifyEvent: function(eventName, eventData)	{
 93 		var count = 0;
 94 		for(var i=0;i<this.observers.length;i++)	{
 95 			try {
 96 				var result = this.observers[i].notify(eventName, eventData);
 97 				if (result == true)	{
 98 					count++;
 99 				}
100 			} catch (err)	{
101 				log(err);
102 			}
103 		}
104 	},
105 	
106 	toString: function() {
107 		return "Subject";
108 	}
109 });
