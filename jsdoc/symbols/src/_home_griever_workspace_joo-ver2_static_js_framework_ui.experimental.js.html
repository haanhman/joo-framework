155

165 /**
166  * @class A movie clip is a sprite with custom animation, it also supports script.
167  * <p>It supports additional configuration parameters</p>
168  * <ul>
169  * 	<li><code>data</code> The animation data</li>
170  * </ul>
171  * @augments JOOSprite
172  */
173 JOOMovieClip = JOOSprite.extend({
174 	
175 	setupBase: function(config) {
176 		this.skippedAnimation = ['position'];
177 		this.intervals = Array();
178 		this._super(config);
179 	},
180 	
181 	setupDomObject: function(config) {
182 		this._super(config);
183 		this.generateData(config.data);
184 	},
185 	
186 	generateData: function(data) {
187 		this.animationsMeta = Array();
188 		this.data = data;
189 		this.objectDefs = {};
190 		this.objects = {};
191 		var objects = data.objects;
192 		for(var i in objects) {
193 			this.objectDefs[objects[i].name] = objects[i];
194 		}
195 		this.buildStage();
196 		this.buildAnimations();
197 	},
198 	
199 	buildStage: function() {
200 		var stageDef = this.data.stage;
201 		this.animStage = new window[stageDef.className](stageDef.config);
202 		var children = stageDef.children;
203 		for(var i in children) {
204 			var obj = this.buildChildren(children[i]);
205 			this.animStage.addChild(obj);
206 		}
207 		this.addChild(this.animStage);
208 	},
209 	
210 	buildChildren: function(child) {
211 		var objDef = this.objectDefs[child.ref];
212 		if (objDef == undefined) throw child.ref+" is not undefined";
213 		var obj = undefined;
214 		obj = new window[objDef.className](objDef.config);
215 		// build attribute
216 		for(var i in objDef.attributes){
217 			obj.setAttribute(i,objDef.attributes[i]);
218 		}
219 		obj.setStyle("display","none");
220 		if(!obj.getStyle("position")){ obj.setStyle("position","absolute"); }
221 		if (objDef.type == "composition") {
222 			//obj.setLayout('absolute');
223 			for (var i in objDef.children) {
224 				var _child = this.buildChildren(objDef.children[i]);
225 				obj.addChild(_child);
226 			}
227 		}
228 		if (child.id) {
229 			this.objects[child.id] = obj;
230 		}
231 		if( obj.play ){
232 			obj.play();
233 		}
234 		this[objDef.varName] = obj;
235 		return obj;
236 	},
237 	
238 	buildAnimations: function() {
239 		//this.animations = this.data.animations;
240 		this.actions = {};
241 		for(var i in this.data.actions) {
242 			this.actions[this.data.actions[i].name] = this.data.actions[i];
243 		}
244 		this.animations = {};
245 		for(var i in this.data.animations) {
246 			var delay = this.data.animations[i].delay;
247 			this.animations[delay] = this.animations[delay] || new Array();
248 			this.animations[delay].push(this.data.animations[i]);
249 		}
250 		this.horizontalFramesNo = this.data.frames;
251 		this.verticalFramesNo = 1;
252 	},
253 	
254 	_stripOldAnimationsMeta: function() {
255 		var frame = this.currentFrame;
256 		for(var i=this.animationsMeta.length-1; i>=0; i--) {
257 			if (frame - this.animationsMeta[i].delay >= this.animationsMeta[i].duration) {
258 				this.animationsMeta.splice(i, 1);
259 			}
260 		}
261 	},

282 	play: function() {
283 		if (this.played) {
284 			this._super();
285 			return;
286 		}
287 		var _self = this;
288 		setTimeout(function() {
289 			_self.played = true;
290 			_self.play();
291 		}, 10);
292 	},
293 	
294 	onFrame: function(frame) {
295 		var animations = this.animations[frame];
296 		if (animations) {
297 			for(var i in animations) {
298 				if (animations[i].script_ref)
299 					this.callScript(animations[i]);
300 				else
301 					this.playAnimation(animations[i], 0);
302 			}
303 		}
304 		//for(var i in this.animations) {
305 			//var animation = this.animations[i];
306 			//this.playAnimation(animation, 0);
307 		//}
308 	},
309 	
310 	callScript: function(animation) {
311 		var fn = window[animation.script_ref];
312 		if (fn) {
313 			var args = animation.script_args;
314 			if (args == undefined || args == "")
315 				args = "[]";
316 			try {
317 				fn.apply(this, JSON.parse(args));
318 			} catch (e) {
319 				log(e);
320 			}
321 		}
322 	},
323 	
324 	playAnimation: function(animation, time) {
325 		if (time >= animation.loop && animation.loop != -1) return;
326 		var objRef = animation.object_ref.split('#');
327 		var obj = this.objects[objRef[0]];
328 		for(var i=1; i<objRef.length; i++) {
329 			obj = obj.children[objRef[i]];
330 		}
331 		var actions = this.actions[animation.action_ref];
332 		var _self = this;
333 		obj.setCSS3Style('transition-timing-function', 'linear');
334 		this.animationsMeta.push(new JOOAnimationData({
335 			object: obj,
336 			delay: animation.delay,
337 			start: actions.start, 
338 			end: actions.end, 
339 			duration: actions.duration
340 		}));
341 		_self.doPlayAnimation([obj, actions, time, animation]);
342 		
343 		//setTimeout(function(args) {
344 			//_self.doPlayAnimation(args);
345 			// var i = setInterval(function(args) {
346 			// 	_self.doPlayAnimation(args);
347 			// }, args[1].interval, args);
348 			// _self.intervals.push(i);
349 		//}, animation.delay, [obj, actions, time, animation]);
350 	},
351 	
352 	doPlayAnimation: function(args) {
353 		var _obj = args[0];
354 		var _actions = args[1];
355 		_obj.setCSS3Style('transition-property', '');
356 		_obj.setCSS3Style('transition-duration', '');
357 		var keys = this.setStyles(_obj, _actions.start);
358 		_obj.getStyle('-webkit-transform');
359 		
360 		var duration = _actions.duration / this.framerate * 1000;
361 		_obj.setCSS3Style('transition-duration', duration+'ms');
362 		_obj.setCSS3Style('transition-property', keys.join(','));
363 		//console.log(_obj.className,JSON.stringify(_obj.getStyle('-webkit-transition-duration')));
364 		
365 		this.setStyles(_obj, _actions.end);
366 	},
367 	
368 	setStyles: function(obj, actions) {
369 		var styles = actions.split(';');
370 		var keys = Array();
371 		for(var i in styles) {
372 			var kv = styles[i].split(':');
373 			if (kv.length < 2) continue;
374 			var k = kv[0].trim();
375 			var v = kv[1].trim();
376 			keys.push(k);
377 			obj.setStyle(k, v);
378 		}
379 		return keys;
380 	},
381 	
382 	dispose: function() {
383 		for(var i in this.intervals){
384 			clearInterval(this.intervals[i]);
385 		}
386 		this._super();
387 	}
388 });
