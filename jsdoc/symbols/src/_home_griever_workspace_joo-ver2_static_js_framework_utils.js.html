  8  * Generate an unique ID
  9  * @param {String} type the type used for generation
 10  * @returns an unique ID
 11  */
 12 function generateId(type)	{
 13 	if (!isPropertySet(utils_items, type))	{
 14 		setProperty(utils_items, type, 0);
 15 	}
 16 	setProperty(utils_items, type, getProperty(utils_items, type)+1);
 17 	return type+"-"+getProperty(utils_items, type);
 18 }

 73 function trimOff(txt, maxLen)	{
 74 	txt = txt.trim();
 75 	if (txt.length > maxLen)	{
 76 		txt = txt.substr(0, maxLen);
 77 		var lastIndexOf = txt.lastIndexOf(' ');
 78 		if (lastIndexOf != -1)
 79 			txt = txt.substr(0, lastIndexOf)+'...';
 80 	}
 81 	return txt;
 82 }

137 		}

139 	for(var i=0; i<callstack.length; i++) {
140 		console.log(callstack[i]);

144 MathUtil = {
145 	
146 	getDistance: function(s, d) {
147 		return Math.sqrt(Math.pow(d.y - s.y, 2) + Math.pow(d.x - s.x, 2));
148 	},
149 	
150 	getAngle: function(s, d, deg) {
151 		var dx = d.x-s.x;
152 		var dy = d.y-s.y;
153 		var atan = Math.atan2(dy, Math.abs(dx));
154 		if (dx < 0) {
155 			atan = Math.PI - atan;
156 		}
157 		if (deg == undefined)
158 			return atan;
159 		return atan*180/Math.PI;
160 	}
161 };

163 function getPositionInRotatedcoordinate(old, angle) { //angle in radian
164 	var a = Math.sqrt(Math.pow(old.x, 2) + Math.pow(old.y, 2));
165 	var originAngle = Math.atan2(old.y, old.x);
166 	return {
167 		x: a*Math.cos(angle-originAngle),
168 		y: -a*Math.sin(angle-originAngle)
169 	};
170 }

172 function getPositionFromRotatedCoordinate(pos, angle, coef) { // angle in radian
173 	if (!coef || !(coef.x && coef.y)) coef = {
174 		x: 0,
175 		y: 0
176 	};
177 	var a = Math.sqrt(Math.pow(pos.x, 2) + Math.pow(pos.y, 2));
178 	var originAngle = Math.atan2(pos.y, pos.x);
179 	return {
180 		x: a*Math.cos(angle+originAngle) + coef.x,
181 		y: a*Math.sin(angle+originAngle) + coef.y
182 	};
183 }
