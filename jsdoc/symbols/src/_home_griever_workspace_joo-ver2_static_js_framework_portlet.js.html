 21 /**
 22  * @class An interface for all portlets.
 23  * A portlet is a pluggable UI components that is managed
 24  * and rendered by JOO framework. A portlet is independent
 25  * from the rest of the application.
 26  * @interface
 27  */
 28 PortletInterface = InterfaceImplementor.extend({
 29 	implement: function(obj)	{
 30 		
 31 		obj.prototype.toString = obj.prototype.toString || function() {
 32 			return this.getName();
 33 		};
 34 		
 35 		/**
 36 		 * Get the name of the portlet. By default, it equals to the className of the
 37 		 * portlet.
 38 		 * @methodOf PortletInterface#
 39 		 * @name getName
 40 		 * @returns {String} The name of the portlet.
 41 		 */
 42 		obj.prototype.getName = obj.prototype.getName || function()	{
 43 			return this.className;
 44 		};
 45 		
 46 		/**
 47 		 * Called automatically by JOO framework when the portlet is initialized.
 48 		 * @methodOf PortletInterface#
 49 		 * @name onBegin
 50 		 */
 51 		obj.prototype.onBegin = obj.prototype.onBegin || function(){};
 52 		
 53 		/**
 54 		 * Called automatically by JOO framework when the portlet is loaded into DOM.
 55 		 * @methodOf PortletInterface#
 56 		 * @name run
 57 		 */
 58 		obj.prototype.run = obj.prototype.run || function(){};
 59 		
 60 		/**
 61 		 * Called automatically by JOO framework when the portlet is reloaded.
 62 		 * @methodOf PortletInterface#
 63 		 * @name onReloadPage
 64 		 */
 65 		obj.prototype.onReloadPage = obj.prototype.onReloadPage || function()	{};
 66 		
 67 		/**
 68 		 * Called automatically by JOO framework when the portlet is no longer needed.
 69 		 * @methodOf PortletInterface#
 70 		 * @name onEnd
 71 		 */
 72 		obj.prototype.onEnd = obj.prototype.onEnd || function(){};
 73 		
 74 		/**
 75 		 * Get the placeholder (container) of the portlet.
 76 		 * @methodOf PortletInterface#
 77 		 * @name getPortletPlaceholder
 78 		 * @returns {PortletPlaceholder} the placeholder of the portlet
 79 		 */
 80 		obj.prototype.getPortletPlaceholder = obj.prototype.getPortletPlaceholder || function()	{
 81 			return this.placeholder;
 82 		};
 83 		
 84 		/**
 85 		 * Change the placeholder (container) of the portlet.
 86 		 * This method is not intended to be used by developers.
 87 		 * @methodOf PortletInterface#
 88 		 * @name setPortletPlaceholder
 89 		 * @param {PortletPlaceholder} plhd the new placeholder of the portlet
 90 		 */
 91 		obj.prototype.setPortletPlaceholder = obj.prototype.setPortletPlaceholder || function(plhd)	{
 92 			this.placeholder = plhd;
 93 		};
 94 		
 95 		/**
 96 		 * Get the page instance
 97 		 * @methodOf PortletInterface#
 98 		 * @name getPage
 99 		 * @returns {Page} the page instance
100 		 */
101 		obj.prototype.getPage = obj.prototype.getPage || function()	{
102 			return SingletonFactory.getInstance(Page);
103 		};
104 		
105 		/**
106 		 * Get the init parameters of the portlet. These parameters are
107 		 * usually configured in a <code>layout.txt</code>
108 		 * @methodOf PortletInterface#
109 		 * @name getInitParameters
110 		 * @param {Page} the page instance
111 		 */
112 		obj.prototype.getInitParameters = obj.prototype.getInitParameters || function()	{
113 			if (this.initParams == undefined)
114 				this.initParams = Array();
115 			return this.initParams;
116 		};
117 		
118 		/**
119 		 * Change the init parameters. This method is not intended to be used
120 		 * by developers.
121 		 * @methodOf PortletInterface#
122 		 * @name setInitParameters 
123 		 * @param {Object} params the init parameters
124 		 */
125 		obj.prototype.setInitParameters = obj.prototype.setInitParameters || function(params)	{
126 			this.initParams = params;
127 		};
128 		
129 		/**
130 		 * Get the current request
131 		 * @methodOf PortletInterface#
132 		 * @name getRequest
133 		 * @param {Request} the current request
134 		 */
135 		obj.prototype.getRequest = obj.prototype.getRequest || function()	{
136 			return this.getPage().getRequest();
137 		};
138 		
139 		obj.prototype.requestForMatchingEffectiveResource = obj.prototype.requestForMatchingEffectiveResource || function(resourceName, condition)	{
140 			var app = SingletonFactory.getInstance(Application);
141 			var rm = app.getResourceManager();
142 			return rm.requestForCustomResource("#effective-area #"+this.getName()+"-"+resourceName+" "+condition);
143 		};
144 		
145 		/**
146 		 * Get the portlet resource. This resource resides in the portlet template
147 		 * and is not visible to users.
148 		 * @methodOf PortletInterface#
149 		 * @name getPortletResource
150 		 * @param resourceName the name (or ID) of the resource
151 		 * @returns {Resource} the portlet (means template) resource with matching name
152 		 */
153 		obj.prototype.getPortletResource = obj.prototype.getPortletResource || function(resourceName)	{
154 			var app = SingletonFactory.getInstance(Application);
155 			var rm = app.getResourceManager();
156 			return rm.requestForCustomResource("#"+this.getName()+"-RootData #"+this.getName()+"-"+resourceName);
157 		};
158 		
159 		/**
160 		 * Get the portlet DOM resource. This resource resides in the portlet rendered
161 		 * content and is visible to users.
162 		 * @methodOf PortletInterface#
163 		 * @name getDomResource
164 		 * @param resourceName the name of the resource
165 		 * @returns {Resource} the DOM (means rendered) resource with matching name
166 		 */
167 		obj.prototype.getDomResource = obj.prototype.getDomResource || function(resourceName)	{
168 			var app = SingletonFactory.getInstance(Application);
169 			var rm = app.getResourceManager();
170 			return rm.requestForCustomResource("#effective-area #"+this.getName()+"-"+resourceName);
171 		};
172 		
173 		/**
174 		 * Get the <code>HTML ID</code> of a resource by its name.
175 		 * @methodOf PortletInterface#
176 		 * @name getResourceID
177 		 * @param resourceName the name of the resource
178 		 * @returns {String} the ID of the resource with matching name
179 		 */
180 		obj.prototype.getResourceID = obj.prototype.getResourceID || function(resourceName)	{
181 			return this.getName()+"-"+resourceName;
182 		};
183 		
184 		/**
185 		 * Get a localized text.
186 		 * @methodOf PortletInterface#
187 		 * @name getLocalizedText
188 		 * @param resourceName the name of the text
189 		 * @returns {String} the localized text
190 		 */
191 		obj.prototype.getLocalizedText = obj.prototype.getLocalizedText || function(resourceName)	{
192 			var app = SingletonFactory.getInstance(Application);
193 			var rm = app.getResourceManager();
194 			var res = rm.requestForResource(this.getName(), "Text"+resourceName);
195 			if (res == undefined)
196 				return undefined;
197 			return res.html();
198 		};
199 		
200 		/**
201 		 * Get a localized message. A message can be parameterized.
202 		 * @methodOf PortletInterface#
203 		 * @name getLocalizedMessage
204 		 * @param resourceName the name of the message
205 		 * @returns {String} the localized message
206 		 */
207 		obj.prototype.getLocalizedMessage = obj.prototype.getLocalizedMessage || function(resourceName)	{
208 			var app = SingletonFactory.getInstance(Application);
209 			var rm = app.getResourceManager();
210 			var res = rm.requestForResource(this.getName(), "Message"+resourceName);
211 			if (res == undefined)
212 				return undefined;
213 			var unresolved = res.html();
214 			
215 			var resolved = unresolved;
216 			//resolved string pattern
217 			for(var i=1;i<arguments.length;i++)	{
218 				resolved = resolved.replace("%"+i, arguments[i]);
219 			}
220 			return resolved;
221 		};
222 	}

225 /**
226  * @class An interface for all components which need rendering.
227  * The <code>RenderInterface</code> is commonly used in 
228  * user-defined <code>Portlet</code>. Note that a component can
229  * have multiple extra views besides the main view. In this case,
230  * developers should use <code>renderView</code> method.
231  * @interface
232  */
233 RenderInterface = InterfaceImplementor.extend({
234 	
235 	onModelChange: function() {
236 		
237 	},
238 	
239 	implement: function(obj)	{
240 		/**
241 		 * Render the component using microtemplating mechanism.
242 		 * The component must supply the following:
243 		 * <p>
244 		 * 	<ul>
245 		 *  	<li>A <code>viewId</code> or implement <code>getName</code> method</li>
246 		 *  	<li>An optional <code>model</code> which is a Javascript object</li>
247 		 *  	<li>A template, which must exists in DOM before this method is called.
248 		 *  		The template should be a <code>script</code> element, with 
249 		 *  		<i><code>text/html</code></i> <code>type</code> attributes.
250 		 *  		The <code>id</code> of the template is the <code>viewId</code>
251 		 *  		followed by "View".
252 		 *  		<br />For example, suppose the <code>viewId</code> of the component
253 		 *  		is MyComponent, then the <code>id</code> should be MyComponentView.
254 		 *  	</li>
255 		 *  </ul>
256 		 * </p>
257 		 * @name render
258 		 * @methodOf RenderInterface#
259 		 * @returns {String} the rendered content of the component
260 		 */
261 		obj.prototype.render = obj.prototype.render || function(){
262 			this.viewId = this.viewId || this.getName()+"View";
263 			this.model = this.model || JOOModel.from({});
264 //			if(this.viewId == undefined || this.model == undefined){
265 //				throw "No viewId or model for rendering";
266 //			}
267 			return tmpl(this.viewId, this.model);
268 		};
269 
270 		/**
271 		 * Render a specific view the component using microtemplating mechanism.
272 		 * The component must supply the following:
273 		 * <p>
274 		 * 	<ul>
275 		 *  	<li>A <code>viewId</code> or implement <code>getName</code> method</li>
276 		 *  	<li>A view template, which must exists in DOM before this method is called.
277 		 *  		The template should be a <code>script</code> element, with 
278 		 *  		<i><code>text/html</code></i> <code>type</code> attributes.
279 		 *  		The <code>id</code> of the template is the <code>viewId</code>
280 		 *  		followed by "-" and the <code>view</code> parameters.
281 		 *  		<br />For example, suppose the <code>viewId</code> of the component
282 		 *  		is MyComponent, then calling <code>this.renderView("FirstView", {})</code>
283 		 *  		inside the component will render the template with <code>id</code> 
284 		 *  		MyComponent-FirstView.
285 		 *  	</li>
286 		 *  </ul>
287 		 * </p>
288 		 * @methodOf RenderInterface#
289 		 * @name renderView
290 		 * @returns {String} the rendered view of the component
291 		 */
292 		obj.prototype.renderView = obj.prototype.renderView || function(view, model)	{
293 			return tmpl((this.viewId || this.getName())+"-"+view, model);
294 		};
295 		
296 		/**
297 		 * Display and bind the model to the view.
298 		 * @methodOf RenderInterface#
299 		 * @name displayAndBind
300 		 */
301 		obj.prototype.displayAndBind = obj.prototype.displayAndBind || function()	{
302 			var _self = this;
303 			if (this.model) {
304 				this.model.addEventListener('change', function() {
305 					_self.getPortletPlaceholder().paintCanvas(_self.render());
306 				});
307 			}
308 			this.getPortletPlaceholder().paintCanvas(this.render());
309 		};
310 	}
311 });

313 PortletPlaceholder = Class.extend(
314 /** @lends PortletPlaceholder# */		
315 {
316 
317 	/**
318 	 * @class A placeholder to store a single portlet.
319 	 * It acts as a bridge between Portlet and {@link PortletCanvas}
320 	 * @augments Class
321 	 * @param canvas the portlet canvas
322 	 * @constructs
323 	 */
324 	init: function(canvas)	{
325 		this.canvas = canvas;
326 	},
327 	
328 	/**
329 	 * Add an object to the canvas
330 	 * @param {Object} object the object to be added
331 	 */
332 	addToCanvas: function(object)	{
333 		this.canvas.addChild(object);
334 	},
335 	
336 	/**
337 	 * Clear everything and repaint the canvas
338 	 * @param {String} html the HTML data to be painted
339 	 */
340 	paintCanvas: function(html)	{
341 		this.canvas.repaint(html);
342 		var subject = SingletonFactory.getInstance(Subject);
343 		subject.notifyEvent('HtmlUpdated');
344 	},
345 	
346 	/**
347 	 * Append to the canvas
348 	 * @param {String} html the HTML data to be appended
349 	 */
350 	drawToCanvas: function(html)	{
351 		this.canvas.paint(html);
352 		var subject = SingletonFactory.getInstance(Subject);
353 		subject.notifyEvent('HtmlUpdated');
354 	},
355 	
356 	/**
357 	 * Access the underlying canvas
358 	 * @returns {PortletCanvas} the portlet canvas
359 	 */
360 	getCanvas: function()	{
361 		return this.canvas;
362 	},
363 	
364 	toString: function() {
365 		return "PortletPlaceholder";
366 	}

369 PortletContainer = Class.extend(
370 /** @lends PortletContainer# */
371 {
372 	/**
373 	 * @class A container which maintains and controls multiple portlets
374 	 * @singleton
375 	 * @augments Class
376 	 * @constructs
377 	 */
378 	init: function()	{
379 		if(PortletContainer.singleton == undefined){
380 			throw "Singleton class";
381 			return undefined;
382 		}
383 		this.portlets = Array();
384 	},
385 	
386 	/**
387 	 * Add a portlet to this container and initialize it
388 	 * @param {PortletInterface} portlet the portlet to be added
389 	 * @param {Object} item portlet metadata
390 	 */
391 	addPortlet: function(portlet, item)	{
392 		var portletMeta = {};
393 		for(var i in item)	{
394 			portletMeta[i] = item[i];
395 		}
396 		portletMeta.portlet = portlet;
397 		if (portletMeta.order == undefined)
398 			portletMeta.order = '';
399 		portletMeta.loaded = false;
400 		this.portlets.push(portletMeta);
401 		try 
402 		{
403 			portlet.onBegin();
404 		} 
405 		catch (err)	{
406 			log(err);
407 		}
408 	},
409 	
410 	/**
411 	 * Move the portlet to another position
412 	 * @param {Object} portletMeta the metadata associated with the portlet to be moved
413 	 * @param {String} newPosition the new position, which is the <code>id</code>
414 	 * of a DOM element
415 	 */
416 	movePortlet: function(portletMeta, newPosition)	{
417 		var portletPosition = new Stage({id: newPosition});
418 		var portletCanvas = new Stage({id: portletMeta.portlet.getPortletPlaceholder().getCanvas().id});
419 		this.attachPortletHtml(portletPosition, portletCanvas, portletMeta);
420 	},
421 	
422 	/**
423 	 * Load all active portlets, execute them synchronously.
424 	 */
425 	loadPortlets: function()	{
426 		for(var i=0;i<this.portlets.length;i++)	{
427 			var portletMeta = this.portlets[i];
428 			if (portletMeta.active == true && !portletMeta.loaded)	{
429 				this.activatePortlet(portletMeta);
430 				portletMeta.loaded = true;
431 			}
432 		}
433 	},
434 	
435 	/**
436 	 * Get all portlets
437 	 * @returns {Array} All loaded portlets
438 	 */
439 	getPortlets: function()	{
440 		return this.portlets;
441 	},
442 	
443 	/**
444 	 * Get portlet metadata using the portlet's name
445 	 * @param {String} name the portlet's name
446 	 */
447 	getPortletMetaByName: function(name)	{
448 		return this.portlets.map(function(portlet) {
449 			if (portlet.portlet.getName() == name)
450 				return portlet;
451 		});
452 	},
453 	
454 	/**
455 	 * Get portlet metadata using the portlet's name
456 	 * @param {String} name the portlet's name
457 	 */
458 	getPortletMetaById: function(id)	{
459 		for(var i=0; i<portlets.length; i++) {
460 			if (portlet.id == id)
461 				return portlet;
462 		}
463 	},
464 	
465 	/**
466 	 * Remove portlet at the specified position
467 	 * @param {String} position the position of the portlet to be removed
468 	 */
469 	removePortlet: function(position)	{
470 		var portletMeta = this.portlets[position];
471 		if (portletMeta != undefined)	{
472 			this.portlets.splice(position,1);
473 			portletMeta.portlet.onEnd();
474 			if (portletMeta.portlet.getPortletPlaceholder())	{
475 				//console.log("dispose canvas of portlet: "+portletMeta.portlet.getName());
476 				portletMeta.portlet.getPortletPlaceholder().getCanvas().dispose();
477 			}
478 		}
479 	},
480 	
481 	attachPortletHtml: function(portletPosition, portletCanvas, portletMeta)	{
482 		var jPortletCanvas = portletPosition.access();
483 		var canvases = jPortletCanvas.find('.portlet.portlet-canvas');
484 
485 		var found = false;
486 		for(var i=0;i<canvases.length;i++)	{
487 			var canvasI = canvases[i];
488 			if ($(canvasI).attr('order') > portletMeta.order)	{
489 				portletPosition.addChildBeforePosition(portletCanvas, canvasI);
490 				found = true;
491 				break;
492 			}
493 		}
494 		
495 		if (found == false)	{
496 			portletPosition.addChild(portletCanvas);
497 		}
498 		portletCanvas.setAttribute('order', portletMeta.order);
499 	},
500 	
501 	/**
502 	 * Activate a portlet.
503 	 * @param {Object} portletMeta the metadata of the portlet to be activated
504 	 */
505 	activatePortlet: function(portletMeta)	{
506 		var portlet = portletMeta.portlet;
507 		if (portletMeta.loaded)	{
508 			return;
509 		}
510 		var portletPosition = new Stage({id: portletMeta.position});
511 		var portletCanvas = new PortletCanvas(portlet.getName());
512 		this.attachPortletHtml(portletPosition, portletCanvas, portletMeta);
513 		portletCanvas.setAttribute('portlet', portlet.getName());
514 		var portletPlaceholder = new PortletPlaceholder(portletCanvas);
515 		portlet.setPortletPlaceholder(portletPlaceholder);
516 		portletMeta.loaded = true;
517 		try 
518 		{
519 			portlet.run();
520 		} 
521 		catch (err)	{
522 			log(err);
523 		}
524 	},
525 	
526 	/**
527 	 * Deactivate a portlet.
528 	 * @param {Object} portletMeta the metadata of the portlet to be deactivated
529 	 */
530 	deactivatePortlet: function(portletMeta)	{
531 		var portlet = portletMeta.portlet;
532 		if (!portletMeta.loaded)	{
533 			return;
534 		}
535 		portletMeta.loaded = false;
536 		if (portlet.getPortletPlaceholder())	{
537 			portlet.getPortletPlaceholder().paintCanvas('');
538 		}
539 	},
540 	
541 	toString: function() {
542 		return "PortletContainer";
543 	}
544 });
