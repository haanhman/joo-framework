1 /**
  2  * @class A desktop-style dialog. It has a title bar and a content pane.
  3  * <p>It supports additional configuration parameters</p>
  4  * <ul>
  5  * 	<li><code>title</code> The title of the dialog</li>
  6  * </ul>
  7  * @augments UIComponent
  8  * @implements DraggableInterface
  9  */
 10 JOODialog = UIComponent.extend(
 11 /** @lends JOODialog# */
 12 {
 13 	setupDomObject: function(config) {
 14 		this._super(config);
 15 
 16 		this.titleBar = new Sketch();
 17 		this.contentPane = new Sketch();
 18 		this.titleBar.access().addClass('joo-dialog-title');
 19 		this.contentPane.access().addClass('joo-dialog-content');
 20 
 21 		this.addChild(this.titleBar);
 22 		this.addChild(this.contentPane);
 23 		
 24 		var _self = this;
 25 		var closeBtn = new JOOCloseButton({absolute: true});
 26 		closeBtn.onclick = function() {
 27 			_self.close();
 28 		};
 29 		var label = new JOOLabel();
 30 		this.titleBar.addChild(label);
 31 		this.titleBar.addChild(closeBtn);
 32 		
 33 		if (config.title != undefined)
 34 			this.setTitle(config.title);
 35 		
 36 		this.draggable({handle: this.titleBar.access()});
 37 		this.startDrag();
 38 		this.addEventListener('stageUpdated', function() {
 39 			this.afterAdded();
 40 		});
 41 	},
 42 	
 43 	/**
 44 	 * Make the dialog centered by the screen.
 45 	 */
 46 	center: function() {
 47 		this.setLocation(($(window).width() - this.access().outerWidth())/2, ($(window).height()-this.access().outerHeight())/2);
 48 	},
 49 	
 50 	afterAdded: function() {
 51 		var modal = this.config.modal || false;
 52 		if (modal) {
 53 			this.modalSketch = new Sketch();
 54 			this.modalSketch.access().addClass('joo-modal-dialog');
 55 			this.stage.addChild(this.modalSketch, this.access());
 56 			this.modalSketch.setStyle('z-index', parseInt(this.getStyle('z-index')-1));
 57 			this.modalSketch.access().show();
 58 		}
 59 	},
 60 	
 61 	/**
 62 	 * Change the title of the dialog.
 63 	 * @param title the new title
 64 	 */
 65 	setTitle: function(title) {
 66 		this.config.title = title;
 67 		this.titleBar.getChildren()[0].setText(title);
 68 	},
 69 	
 70 	/**
 71 	 * Get the current title of the dialog.
 72 	 * @returns {String} the current title
 73 	 */
 74 	getTitle: function() {
 75 		return this.titleBar.getChildren()[0].getText();
 76 	},
 77 	
 78 	/**
 79 	 * Get the content pane of the dialog.
 80 	 * @returns {Sketch} the content pane
 81 	 */
 82 	getContentPane: function() {
 83 		return this.contentPane;
 84 	},
 85 	
 86 	/**
 87 	 * Close the dialog.
 88 	 */
 89 	close: function() {
 90 		this.dispatchEvent('closing');
 91 		if (this.modalSketch != undefined)
 92 			this.modalSketch.selfRemove();
 93 		this.selfRemove();
 94 	},
 95 	
 96 	/**
 97 	 * Show the dialog.
 98 	 */
 99 	show: function() {
100 		this.access().show();
101 	},
102 	
103 	/**
104 	 * Hide the dialog.
105 	 */
106 	hide: function() {
107 		this.access().hide();
108 	},
109 	
110 	toHtml: function() {
111 		return "<div></div>";
112 	}
113 }).implement(DraggableInterface);

179 /**
180  * @class A slider icon, used in JOOSlider.
181  * @augments Sketch
182  * @implements DraggableInterface
183  */
184 SliderIcon = Sketch.extend({
185 	
186 	setupBase: function(config) {
187 		this._super(config);
188 	},
189 	
190 	setupDomObject: function(config) {
191 		this._super(config);
192 		this.setWidth(18);
193 		this.setHeight(18);
194 		
195 		var _self = this;
196 		this.draggable({containment:'parent'});
197 		
198 		this.addEventListener("mousedown", function(e) {
199 			this.enable = true;
200 			$(window).bind("mousemove", {_self: this}, this.mouseMoveHandler);
201 			this.addEventListener("mousemove", this.mouseMoveHandler);
202 		});
203 		this.addEventListener("mouseup", function(e) {
204 			this.mouseUpHandler(e);
205 		});
206 		
207 		$(window).bind("mouseup", function(e) {
208 			_self.mouseUpHandler(e);
209 		});
210 	},
211 	
212 	mouseUpHandler: function(e) {
213 		this.enable = false;
214 		$(window).unbind("mousemove", this.mouseMoveHandler);
215 		this.removeEventListener("mousemove", this.mouseMoveHandler);
216 	},
217 	
218 	mouseMoveHandler: function(e) {
219 		var _self = e.data ? e.data._self || this : this;
220 		if (_self.enable) {
221 			_self.dispatchEvent('slideChanged', {value: _self.getX(), ispos: "pos"});
222 		}
223 	}
224 }).implement(DraggableInterface);

226 /**
227  * @class A slider, which has a draggable icon and a slide pane.
228  * @augments JOOInput
229  */
230 JOOSlider = JOOInput.extend(
231 /** @lends JOOSlider# */
232 {
233 	setupBase: function(config)	{
234 		this._super(config);
235 	},
236 	
237 	setupDomObject: function(config) {
238 		config.width = config.width || 200;
239 		config.height = 5;
240 		this._super(config);
241 
242 		this.sliderIcon = new SliderIcon();
243 		this.value = config.value || 0;
244 		this.min = config.min || 0;
245 		this.max = config.max || 100;
246 
247 		this.addChild(this.sliderIcon);
248 		
249 		this.addEventListener('stageUpdated', function() {
250 			this.slideTo(this.getValue());
251 		});
252 		
253 		var _self = this;
254 		this.sliderIcon.addEventListener("slideChanged", function(e) {
255 			_self.slideTo(e.value, e.ispos);
256 		});
257 		
258 		this.sliderIcon.addEventListener("mouseover", function(e) {
259 			showtip(new Number(_self.getValue()).toFixed(2));
260 		});
261 		
262 		this.sliderIcon.addEventListener("drag", function(e) {
263 			showtip(new Number(_self.getValue()).toFixed(2));
264 		});
265 		
266 		this.sliderIcon.addEventListener("dragstop", function(e) {
267 			hidetip();
268 		});
269 		
270 		this.sliderIcon.addEventListener("mouseout", function(e) {
271 			hidetip();
272 		});
273 	},
274 	
275 	/**
276 	 * SLide the icon to a specific value.
277 	 * @param {String|Number} value the value of the slider.
278 	 * @param {Boolean} ispos whether the <code>value</code> is position or absolute value.
279 	 */
280 	slideTo: function(value, ispos){
281 		if(ispos){
282 			// position, not value anymore
283 			var posX = value;
284 			value = ((parseFloat(posX)) / (this.getWidth() - 18)) * (this.max - this.min) + this.min;
285 			if(value < this.min){
286 				value = this.min;
287 				posX = 9;
288 				this.sliderIcon.setX(posX);
289 			}
290 		} else {
291 			if (value < this.min) {
292 				value = this.min;
293 			}
294 			if (value > this.max) {
295 				value = this.max;
296 			}
297 			var posX = (value - this.min) / (this.max - this.min) * (this.getWidth() - 18);
298 			this.sliderIcon.setX(posX);
299 		}
300 		
301 		var rate = (value - this.min) / (this.max - this.min);
302 		var lWidth = rate * (this.getWidth() - 18);
303 		var rWidth = (1 - rate) * (this.getWidth() - 18);
304 		this.access().find('.active').css("width", lWidth + "px");
305 		this.access().find('.inactive').css("width", rWidth + "px");
306 		this.access().find("input").val(value);
307 		
308 		this.dispatchEvent('change');
309 	},
310 
311 	/**
312 	 * Change the value of the slider.
313 	 * @param {Number} value the new value
314 	 */
315 	setValue: function(value) {
316 		this.slideTo(value);
317 	},
318 	
319 	/**
320 	 * Get the value of the slider.
321 	 * @returns {Number} the value of the slider
322 	 */
323 	getValue: function()	{
324 		return this.access().find("input").val();
325 	},
326 	
327 	toHtml: function()	{
328 		return "<div><input type='hidden'><div class='joo-slider-bg active'></div><div class='joo-slider-bg inactive'></div></div>";
329 	}
330 });

332 /**
333  * @class A simple color picker. Wrapper of jQuery ColorPicker.
334  * @augments JOOInput
335  */
336 JOOColorPicker = JOOInput.extend(
337 /** @lends JOOColorPicker# */
338 {
339 	setupBase: function(config) {
340 		this._super(config);
341 	},
342 	
343 	setupDomObject: function(config) {
344 		config.width = 18;
345 		config.height = 18;
346 		
347 		this._super(config);
348 		
349 		if(!config.background ) {
350 			config.background = "#FFF";
351 		}
352 		
353 		this.shown = false;
354 		var _cpicker = this;
355 		var c = config.background.substring(1,config.background.length);
356 		this.access().ColorPicker({
357 			flat: true,
358 			onChange: function(){
359 				var hex = arguments[1];
360 				_cpicker.colorPickerIcon.setStyle("background-color", hex);
361 				_cpicker.dispatchEvent('change');
362 			},
363 			color:c
364 		});
365 		this.colorPickerIcon = new Sketch(config);
366 		var colorPanel = this.access().find(".colorpicker");
367 		this.colorPanelId = colorPanel.attr("id");
368 		colorPanel.css("left", "22px");
369 		colorPanel.css("top", "-2px");
370 		
371 		var _self = this;
372 		this.colorPickerIcon.addEventListener("mouseover", function() {
373 			showtip(_self.getValue());
374 		});
375 		
376 		this.colorPickerIcon.addEventListener("mouseout", function() {
377 			hidetip();
378 		});
379 		
380 		this.colorPickerIcon.addEventListener("click", function(e) {
381 			if(_cpicker.shown) {
382 				$("#"+_cpicker.colorPanelId).hide();
383 				_cpicker.shown = false;
384 			} else {
385 				$("#"+_cpicker.colorPanelId).show();
386 				_cpicker.shown = true;
387 			}
388 		});
389 		this.addChild(this.colorPickerIcon);
390 		this.colorPickerIcon.access().addClass('joo-colorpicker-icon');
391 		this.colorPickerIcon.setStyle("background", config.background);
392 	},
393 	
394 	/**
395 	 * Change the value of the picker.
396 	 * @param {String} value the new value
397 	 */
398 	setValue: function(value){
399 		this.colorPickerIcon.setStyle("background-color", value);
400 	},
401 	
402 	/**
403 	 * Get the value of the picker.
404 	 * @returns {String} the picker's value
405 	 */
406 	getValue: function(){
407 		return this.colorPickerIcon.getStyle("background-color");
408 	},
409 	
410 	toHtml: function(){
411 		return "<div></div>";
412 	}
413 });
