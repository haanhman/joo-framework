1 /**
  2  * @class An interface enabling UI Components to be rendered
  3  * using composition
  4  * @interface
  5  */
  6 CompositionRenderInterface = InterfaceImplementor.extend({
  7 	
  8 	implement: function(obj) {
  9 		var _self = this;
 10 		
 11 		/**
 12 		 * Render the UI Component.
 13 		 * @methodOf CompositionRenderInterface#
 14 		 * @name renderUIComposition
 15 		 */
 16 		obj.prototype.renderUIComposition = obj.prototype.renderUIComposition || function() {
 17 			var composition = $($('#UI-'+this.className)[0].innerHTML);
 18 			_self.processElement(this, this, composition[0]);
 19 		};
 20 	},
 21 	
 22 	processElement: function(root, obj, composition) {
 23 		var $composition = $(composition);
 24 		var tagName = composition.tagName.toLowerCase();
 25 		var children = $composition.children();
 26 		var currentObject = obj;
 27 		var config = JOOUtils.getAttributes(composition);
 28 		
 29 		switch(tagName) {
 30 		case joo:composition:
 31 			for(var i in config) {
 32 				var mutator = ExpressionUtils.getMutatorMethod(currentObject, i);
 33 				mutator.call(currentObject, config[i]);
 34 			}
 35 			break;
 36 		case joo:var:
 37 			var varName = $composition.attr('name');
 38 			currentObject = obj[varName];
 39 			break;
 40 		default:
 41 			var className = ClassMapping[tagName.split(':')[1]];
 42 			currentObject = new window[className](config);
 43 		}
 44 		
 45 		var varName = $composition.attr('varName');
 46 		if (varName) {
 47 			root[varName] = currentObject;
 48 		}
 49 		
 50 		for(var i=0; i<children.length; i++) {
 51 			var child = this.processElement(root, currentObject, children[i]);
 52 			currentObject.addChild(child);
 53 		}
 54 		return currentObject;
 55 	}
 56 });
