1 xui.extend({
  2 	/**
  3 	 * Adds more DOM nodes to the existing element list.
  4 	 */
  5 	add: function(q) {
  6 	  [].push.apply(this, slice(xui(q)));
  7 	  return this.set(this.reduce());
  8 	},
  9 
 10 	/**
 11 	 * Pops the last selector from XUI
 12 	 */
 13 	end: function () {	
 14 		return this.set(this.cache || []);	 	
 15 	},
 16   
 17 	/**
 18 	 * Sets the `display` CSS property to `block`.
 19 	 */
 20 	show: function() {
 21 		return this.setStyle('display','block');
 22 	},
 23  
 24   /**
 25 	 * Sets the `display` CSS property to `none`.
 26 	 */
 27 	hide: function() {
 28 		return this.setStyle('display','none');
 29 	},
 30   
 31 	appendTo: function(obj) {
 32 		return obj[0].appendChild(this[0]);
 33 	},
 34   
 35 	width: function() {
 36 		if (this[0] == window)
 37 			return this[0].innerWidth;
 38 		var w = this.css('width');
 39 		if (w.indexOf('px') != -1)
 40 			return w.substring(0, w.length-2);
 41 		return w;
 42 	},
 43 	
 44 	height: function() {
 45 		if (this[0] == window)
 46 			return this[0].innerHeight;
 47 		var h = this.css('height');
 48 		if (h.indexOf('px') != -1)
 49 			return h.substring(0, h.length-2);
 50 		return h;
 51 	},
 52 	
 53 	bind: function(types, fn) {
 54 		var t = types.split(' ');
 55 		for(var i in t) {
 56 			this.on(t[i], fn);
 57 		}
 58 	},
 59 	
 60 	prepend: function(html) {
 61 		return this.html("top", html);
 62 	}
 63 });
