  9 (function($){
 10     $.fn.getStyleObject = function(){
 11         var dom = this.get(0);
 12         var style;
 13         var returns = {};
 14         if(window.getComputedStyle){
 15             var camelize = function(a,b){
 16                 return b.toUpperCase();
 17             };
 18             style = window.getComputedStyle(dom, null);
 19             for(var i=0;i<style.length;i++){
 20                 var prop = style[i];
 21                 var camel = prop.replace(/\-([a-z])/g, camelize);
 22                 var val = style.getPropertyValue(prop);
 23                 returns[camel] = val;
 24             }
 25             return returns;
 26         }
 27         if(dom.currentStyle){
 28             style = dom.currentStyle;
 29             for(var prop in style){
 30                 returns[prop] = style[prop];
 31             }
 32             return returns;
 33         }
 34         return this.css();
 35     };
 36 })(jQuery);
